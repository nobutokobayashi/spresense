89 /****************************************************************************
90  * Name: gran_initialize
91  *
92  * Description:
93  * Set up one granule allocator instance. Allocations will be aligned to
94  * the alignment size (log2align; allocations will be in units of the
95  * granule size (log2gran). Larger granules will give better performance
96  * and less overhead but more losses of memory due to quantization waste.
97  * Additional memory waste can occur from alignment; log2align should be
98  * set to 0 unless you are using the granule allocator to manage DMA
99  * or page-aligned memory and your hardware has specific memory alignment
100  * requirements.
101  *
102  * General Usage Summary. This is an example using the GCC section
103  * attribute to position a DMA heap in memory (logic in the linker script
104  * would assign the section .dmaheap to the DMA memory.
105  *
106  * FAR uint32_t g_dmaheap[DMAHEAP_SIZE] __attribute__((section(.dmaheap)));
107  *
108  * The heap is created by calling gran_initialize. Here the granule size
109  * is set to 64 bytes and the alignment to 16 bytes:
110  *
111  * GRAN_HANDLE handle = gran_initialize(g_dmaheap, DMAHEAP_SIZE, 6, 4);
112  *
113  * Then the GRAN_HANDLE can be used to allocate memory (There is no
114  * GRAN_HANDLE if CONFIG_GRAN_SINGLE=y):
115  *
116  * FAR uint8_t *dma_memory = (FAR uint8_t *)gran_alloc(handle, 47);
117  *
118  * The actual memory allocates will be 64 byte (wasting 17 bytes) and
119  * will be aligned at least to (1 << log2align).
120  *
121  * NOTE: The current implementation also restricts the maximum allocation
122  * size to 32 granules. That restriction could be eliminated with some
123  * additional coding effort.
124  *
125  * Input Parameters:
126  * heapstart - Start of the granule allocation heap
127  * heapsize - Size of heap in bytes
128  * log2gran - Log base 2 of the size of one granule. 0->1 byte,
129  * 1->2 bytes, 2->4 bytes, 3->8 bytes, etc.
130  * log2align - Log base 2 of required alignment. 0->1 byte,
131  * 1->2 bytes, 2->4 bytes, 3->8 bytes, etc. Note that
132  * log2gran must be greater than or equal to log2align
133  * so that all contiguous granules in memory will meet
134  * the minimum alignment requirement. A value of zero
135  * would mean that no alignment is required.
136  *
137  * Returned Value:
138  * On success, a non-NULL handle is returned that may be used with other
139  * granule allocator interfaces.
140  *
141  ****************************************************************************/
142

146 #else

172 /****************************************************************************
173  * Name: gran_reserve
174  *
175  * Description:
176  * Reserve memory in the granule heap. This will reserve the granules
177  * that contain the start and end addresses plus all of the granules
178  * in between. This should be done early in the initialization sequence
179  * before any other allocations are made.
180  *
181  * Reserved memory can never be allocated (it can be freed however which
182  * essentially unreserves the memory).
183  *
184  * Input Parameters:
185  * handle - The handle previously returned by gran_initialize
186  * start - The address of the beginning of the region to be reserved.
187  * size - The size of the region to be reserved
188  *
189  * Returned Value:
190  * None
191  *
192  ****************************************************************************/
193
