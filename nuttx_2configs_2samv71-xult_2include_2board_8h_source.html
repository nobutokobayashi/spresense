99 /* PMC master clock register settings.
100  *
101  * BOARD_PMC_MCKR_CSS - The source of main clock input. This may be one of:
102  *
103  * PMC_MCKR_CSS_SLOW Slow Clock
104  * PMC_MCKR_CSS_MAIN Main Clock
105  * PMC_MCKR_CSS_PLLA PLLA Clock
106  * PMC_MCKR_CSS_UPLL Divided UPLL Clock
107  *
108  * BOARD_PMC_MCKR_PRES - Source clock pre-scaler. May be one of:
109  *
110  * PMC_MCKR_PRES_DIV1 Selected clock
111  * PMC_MCKR_PRES_DIV2 Selected clock divided by 2
112  * PMC_MCKR_PRES_DIV4 Selected clock divided by 4
113  * PMC_MCKR_PRES_DIV8 Selected clock divided by 8
114  * PMC_MCKR_PRES_DIV16 Selected clock divided by 16
115  * PMC_MCKR_PRES_DIV32 Selected clock divided by 32
116  * PMC_MCKR_PRES_DIV64 Selected clock divided by 64
117  * PMC_MCKR_PRES_DIV3 Selected clock divided by 3
118  *
119  * The prescaler determines (1) the CPU clock and (2) the input into the
120  * second divider that then generates the Master Clock (MCK). MCK is the
121  * source clock of the peripheral clocks.
122  *
123  * BOARD_PMC_MCKR_MDIV - MCK divider. May be one of:
124  *
125  * PMC_MCKR_MDIV_DIV1 Master Clock = Prescaler Output Clock / 1
126  * PMC_MCKR_MDIV_DIV2 Master Clock = Prescaler Output Clock / 2
127  * PMC_MCKR_MDIV_DIV4 Master Clock = Prescaler Output Clock / 4
128  * PMC_MCKR_MDIV_DIV3 Master Clock = Prescaler Output Clock / 3
129  */
130
