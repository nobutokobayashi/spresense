57 /* Definitions of 8-bit socket flags */
58 
59  /* Bits 0-2: Socket state */
60 #define _SF_IDLE 0x00 /* - There is no socket activity */
61 #define _SF_ACCEPT 0x01 /* - Socket is waiting to accept a connection */
62 #define _SF_RECV 0x02 /* - Waiting for recv action to complete */
63 #define _SF_SEND 0x03 /* - Waiting for send action to complete */
64 #define _SF_MASK 0x03 /* - Mask to isolate the above actions */
65 
66 #define _SF_NONBLOCK 0x08 /* Bit 3: Don't block if no data (TCP/READ only) */
67 #define _SF_LISTENING 0x10 /* Bit 4: SOCK_STREAM is listening */
68 #define _SF_BOUND 0x20 /* Bit 5: SOCK_STREAM is bound to an address */
69  /* Bits 6-7: Connection state */
70 #define _SF_CONNECTED 0x40 /* Bit 6: SOCK_STREAM is connected */
71 #define _SF_CLOSED 0x80 /* Bit 7: SOCK_STREAM was gracefully disconnected */
72 
73 /* Connection state encoding:
74  *
75  * _SF_CONNECTED==1 && _SF_CLOSED==0 - the socket is connected
76  * _SF_CONNECTED==0 && _SF_CLOSED==1 - the socket was gracefully disconnected
77  * _SF_CONNECTED==0 && _SF_CLOSED==0 - the socket was rudely disconnected
78  */
79

292 /****************************************************************************
293  * Name: psock_send
294  *
295  * Description:
296  * The send() call may be used only when the socket is in a connected state
297  * (so that the intended recipient is known). The only difference between
298  * send() and write() is the presence of flags. With zero flags parameter,
299  * send() is equivalent to write(). Also, send(sockfd,buf,len,flags) is
300  * equivalent to sendto(sockfd,buf,len,flags,NULL,0).
301  *
302  * Parameters:
303  * psock An instance of the internal socket structure.
304  * buf Data to send
305  * len Length of data to send
306  * flags Send flags
307  *
308  * Returned Value:
309  * On success, returns the number of characters sent. On error,
310  * -1 is returned, and errno is set appropriately:
311  *
312  * EAGAIN or EWOULDBLOCK
313  * The socket is marked non-blocking and the requested operation
314  * would block.
315  * EBADF
316  * An invalid descriptor was specified.
317  * ECONNRESET
318  * Connection reset by peer.
319  * EDESTADDRREQ
320  * The socket is not connection-mode, and no peer address is set.
321  * EFAULT
322  * An invalid user space address was specified for a parameter.
323  * EINTR
324  * A signal occurred before any data was transmitted.
325  * EINVAL
326  * Invalid argument passed.
327  * EISCONN
328  * The connection-mode socket was connected already but a recipient
329  * was specified. (Now either this error is returned, or the recipient
330  * specification is ignored.)
331  * EMSGSIZE
332  * The socket type requires that message be sent atomically, and the
333  * size of the message to be sent made this impossible.
334  * ENOBUFS
335  * The output queue for a network interface was full. This generally
336  * indicates that the interface has stopped sending, but may be
337  * caused by transient congestion.
338  * ENOMEM
339  * No memory available.
340  * ENOTCONN
341  * The socket is not connected, and no target has been given.
342  * ENOTSOCK
343  * The argument s is not a socket.
344  * EOPNOTSUPP
345  * Some bit in the flags argument is inappropriate for the socket
346  * type.
347  * EPIPE
348  * The local end has been shut down on a connection oriented socket.
349  * In this case the process will also receive a SIGPIPE unless
350  * MSG_NOSIGNAL is set.
351  *
352  ****************************************************************************/
353 
354 ssize_t psock_send(FAR struct socket *psock, FAR const void *buf, size_t len,
