150 /* GPIO Register Bit Definitions ****************************************************/
151 
152 /* Most registers are laid out simply with one bit per pin */
153 
154 #define GPIO_PIN(n) (1 << (n)) /* Bit n: Pin n, n=0-31 */
155 
156 /* GPIO interrupt configuration register 1/2 */
157 
158 #define GPIO_ICR_INDEX(n) (((n) >> 4) & 1)
159 #define GPIO_ICR_OFFSET(n) (GPIO_ICR1_OFFSET + (GPIO_ICR_INDEX(n) << 2))
160 
161 #define GPIO_ICR_LOWLEVEL 0 /* Interrupt is low-level sensitive */
162 #define GPIO_ICR_HIGHLEVEL 1 /* Interrupt is high-level sensitive */
163 #define GPIO_ICR_RISINGEDGE 2 /* Interrupt is rising-edge sensitive */
164 #define GPIO_ICR_FALLINGEDGE 3 /* Interrupt is falling-edge sensitive */
165 
166 #define GPIO_ICR_SHIFT(n) (((n) & 15) << 1)
167 #define GPIO_ICR_MASK(n) (3 << GPIO_ICR_SHIFT(n))
168 #define GPIO_ICR(i,n) ((uint32_t)(n) << GPIO_ICR_SHIFT(n))
