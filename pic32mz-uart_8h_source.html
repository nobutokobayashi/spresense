200 /* UARTx mode register */
201 
202 #define UART_MODE_STSEL (1 << 0) /* Bit 0: Stop selection 1=2 stop bits */
203 #define UART_MODE_PDSEL_SHIFT (1) /* Bits: 1-2: Parity and data selection */
204 #define UART_MODE_PDSEL_MASK (3 << UART_MODE_PDSEL_SHIFT)
205 # define UART_MODE_PDSEL_8NONE (0 << UART_MODE_PDSEL_SHIFT) /* 8-bit data, no parity */
206 # define UART_MODE_PDSEL_8EVEN (1 << UART_MODE_PDSEL_SHIFT) /* 8-bit data, even parity */
207 # define UART_MODE_PDSEL_8ODD (2 << UART_MODE_PDSEL_SHIFT) /* 8-bit data, odd parity */
208 # define UART_MODE_PDSEL_9NONE (3 << UART_MODE_PDSEL_SHIFT) /* 9-bit data, no parity */
209 #define UART_MODE_BRGH (1 << 3) /* Bit 3: High baud rate enable */
210 #define UART_MODE_RXINV (1 << 4) /* Bit 4: Receive polarity inversion */
211 #define UART_MODE_ABAUD (1 << 5) /* Bit 5: Auto-baud enable */
212 #define UART_MODE_LPBACK (1 << 6) /* Bit 6: UARTx loopback mode select */
213 #define UART_MODE_WAKE (1 << 7) /* Bit 7: Enable wake-up on start bit detect during sleep mode */
214 #define UART_MODE_UEN_SHIFT (8) /* Bits: 8-9: UARTx enable */
215 #define UART_MODE_UEN_MASK (3 << UART_MODE_UEN_SHIFT)
216 # define UART_MODE_UEN_PORT (0 << UART_MODE_UEN_SHIFT) /* UxCTS+UxRTS/UxBCLK=PORTx register */
217 # define UART_MODE_UEN_ENR_CPORT (1 << UART_MODE_UEN_SHIFT) /* UxRTS=enabled; UxCTS=PORTx register */
218 # define UART_MODE_UEN_ENCR (2 << UART_MODE_UEN_SHIFT) /* UxCTS+UxRTS=enabled */
219 # define UART_MODE_UEN_CPORT (3 << UART_MODE_UEN_SHIFT) /* UxCTS=PORTx register */
220 #define UART_MODE_RTSMD (1 << 11) /* Bit 11: Mode selection for ~UxRTS pin */
221 #define UART_MODE_IREN (1 << 12) /* Bit 12: IrDA encoder and decoder enable */
222 #define UART_MODE_SIDL (1 << 13) /* Bit 13: Stop in idle mode */
223 #define UART_MODE_ON (1 << 15) /* Bit 15: UARTx enable */
224 
225 /* UARTx status and control register */
226 
227 #define UART_STA_URXDA (1 << 0) /* Bit 0: Receive buffer data available */
228 #define UART_STA_OERR (1 << 1) /* Bit 1: Receive buffer overrun error status */
229 #define UART_STA_FERR (1 << 2) /* Bit 2: Framing error status */
230 #define UART_STA_PERR (1 << 3) /* Bit 3: Parity error status */
231 #define UART_STA_RIDLE (1 << 4) /* Bit 4: Receiver idle */
232 #define UART_STA_ADDEN (1 << 5) /* Bit 5: Address character detect */
233 #define UART_STA_URXISEL_SHIFT (6) /* Bits: 6-7: Receive interrupt mode selection */
234 #define UART_STA_URXISEL_MASK (3 << UART_STA_URXISEL_SHIFT)
235 # define UART_STA_URXISEL_RECVD (0 << UART_STA_URXISEL_SHIFT) /* Character received */
236 # define UART_STA_URXISEL_RXB50 (1 << UART_STA_URXISEL_SHIFT) /* RX buffer 1/2 full */
237 # define UART_STA_URXISEL_RXB75 (2 << UART_STA_URXISEL_SHIFT) /* RX buffer 3/4 full */
238 #define UART_STA_UTRMT (1 << 8) /* Bit 8: Transmit shift register is empty */
239 #define UART_STA_UTXBF (1 << 9) /* Bit 9: Transmit buffer full status */
240 #define UART_STA_UTXEN (1 << 10) /* Bit 10: Transmit enable */
241 #define UART_STA_UTXBRK (1 << 11) /* Bit 11: Transmit break */
242 #define UART_STA_URXEN (1 << 12) /* Bit 12: Receiver enable */
243 #define UART_STA_UTXINV (1 << 13) /* Bit 13: Transmit polarity inversion */
244 #define UART_STA_UTXISEL_SHIFT (14) /* Bits: 14-15: TX interrupt mode selection bi */
245 #define UART_STA_UTXISEL_MASK (3 << UART_STA_UTXISEL_SHIFT)
246 # define UART_STA_UTXISEL_TXBNF (0 << UART_STA_UTXISEL_SHIFT) /* TX buffer not full */
247 # define UART_STA_UTXISEL_DRAINED (1 << UART_STA_UTXISEL_SHIFT) /* All characters sent */
248 # define UART_STA_UTXISEL_TXBE (2 << UART_STA_UTXISEL_SHIFT) /* TX buffer empty */
249 #define UART_STA_ADDR_SHIFT (16) /* Bits:16-23: Automatic address mask */
250 #define UART_STA_ADDR_MASK (0xff << UART_STA_ADDR_SHIFT)
251 #define UART_STA_ADM_EN (1 << 24) /* Bit 24: Automatic address detect mode enable */
