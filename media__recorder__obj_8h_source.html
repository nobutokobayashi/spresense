195  /* When MP3 encode and fs is 16kHz, 22.05kHz, 24kHz, sample num of
196  * 1au(access unit) is 1152/2 = 576 (It depend on MPEG2 compliant).
197  * Therefore, at first, value is (#1)"SampleNumPerFrame[m_codec_type] / 2".
198  * And sample num of captured and SRC filterd data is to be 576,
199  * return ((#1) * 48000 / m_sampling_rate(Hz)).
200  *
201  * The process below is only for fs is 48kHz, 16kHz.
202  * To correspontd to 32000Hz, 44100Hz..., need conversion process to
203  * sample num per 1au to be 1152.
204  */
205  uint32_t getPcmCaptureSample()
206  {
207  if (m_codec_type == AudCodecMP3 && m_sampling_rate < 32000)
208  {
209  return (SampleNumPerFrame[m_codec_type] / 2 * 48000 /
210  m_sampling_rate);
211  }
212  else if (m_codec_type == AudCodecOPUS)
213  {
214  /* 20ms. */
215 
216  return ((m_sampling_rate / 50) * (48000 / m_sampling_rate));
217  }
218  return SampleNumPerFrame[m_codec_type];
219  }
