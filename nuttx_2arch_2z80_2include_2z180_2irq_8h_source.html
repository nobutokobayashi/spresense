66 /* TRAP Interrupt
67  *
68  * The Z8X180 generates a non-maskable TRAP interrupt when an undefined Op
69  * Code fetch occurs. When a TRAP interrupt occurs the Z8X180 operates as
70  * follows:
71  *
72  * 1. The TRAP bit in the Interrupt TRAP/Control (ITC) register is set to 1.
73  * 2. The current PC (Program Counter) is saved on the stack.
74  * 3. The Z8X180 vectors to logical address 0 (which may or may not be the
75  * same as reset which vectors to physical address 0x00000).
76  *
77  * The state of the UFO (Undefined Fetch Object) bit in ITC allows TRAP
78  * manipulation software to correctly adjust the stacked PC, depending on
79  * whether the second or third byte of the Op Code generated the TRAP. If
80  * UFO is 0, the starting address of the invalid instruction is equal to
81  * the stacked PC-1. If UFO is 1, the starting address of the invalid
82  * instruction is equal to the stacked PC-2.
83  */
84 
85 #define Z180_TRAP (7)
86 
87 /* INT0
88  *
89  * INT0 (only) has 3 different software programmable interrupt response
90  * modes—Mode 0, Mode 1 and Mode 2.
91  *
92  * - INT0 Mode 0. During the interrupt acknowledge cycle, an instruction
93  * is fetched from the data bus (DO–D7) at the rising edge of T3.
94  *
95  * - INT0 Mode 1. The PC is stacked and instruction execution restarts at
96  * logical address 0x0038.
97  *
98  * - INT0 Mode 2. The restart address is obtained by reading the contents
99  * of a table residing in memory. The vector table consists of up to
100  * 128 two-byte restart addresses stored in low byte, high byte order.
101  *
102  * This is similar to normal vector mode interrupts (like INT1 and 2):
103  * The 256-bit table address comes from I, however the lower-order 8
104  * bits of the vector is fetched from the data bus.
105  */
106 
107 #define Z180_INT0 (8)
108 
109 /* Vector Interrupts
110  *
111  * Normal vector interrupts use a vector table with 16 entries (2 bytes
112  * per entry). Each entry holds the address of the interrupt handler.
113  * The vector table address is determined by 11-bits from the I and IL
114  * registers. The vector table must be aligned on 32-byte address
115  * boundaries.
116  */
117

161 /* Common Area 1 holds the code and data that is unique to a particular task
162  * and shared by all pthreads created from that task. Each task will then
163  * have its own copy of struct z180_cbr_s. This structure is created with
164  * a reference count of one when the task is created.
165  *
166  * When the task creates additional threads, the reference count is
167  * incremented and the CBR value is shared. When each thread exits, the
168  * reference count id decremented. When the reference count is decremented,
169  * the physical memory underlying the CBR is finally released.
170  */
171

177 };
