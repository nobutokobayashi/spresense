100 /****************************************************************************
101  * Name: sem_tickwait
102  *
103  * Description:
104  * This function is a lighter weight version of sem_timedwait(). It is
105  * non-standard and intended only for use within the RTOS.
106  *
107  * Parameters:
108  * sem - Semaphore object
109  * start - The system time that the delay is relative to. If the
110  * current time is not the same as the start time, then the
111  * delay will be adjust so that the end time will be the same
112  * in any event.
113  * delay - Ticks to wait from the start time until the semaphore is
114  * posted. If ticks is zero, then this function is equivalent
115  * to sem_trywait().
116  *
117  * Return Value:
118  * Zero (OK) is returned on success. A negated errno value is returned on
119  * failure. -ETIMEDOUT is returned on the timeout condition.
120  *
121  ****************************************************************************/
122 
123 int sem_tickwait(FAR sem_t *sem, systime_t start, uint32_t delay);

165 /****************************************************************************
166  * Name: sem_setprotocol
167  *
168  * Description:
169  * Set semaphore protocol attribute.
170  *
171  * One particularly important use of this function is when a semaphore
172  * is used for inter-task communication like:
173  *
174  * TASK A TASK B
175  * sem_init(sem, 0, 0);
176  * sem_wait(sem);
177  * sem_post(sem);
178  * Awakens as holder
179  *
180  * In this case priority inheritance can interfere with the operation of
181  * the semaphore. The problem is that when TASK A is restarted it is a
182  * holder of the semaphore. However, it never calls sem_post(sem) so it
183  * becomes *permanently* a holder of the semaphore and may have its
184  * priority boosted when any other task tries to acquire the semaphore.
185  *
186  * The fix is to call sem_setprotocol(SEM_PRIO_NONE) immediately after
187  * the sem_init() call so that there will be no priority inheritance
188  * operations on this semaphore.
189  *
190  * Parameters:
191  * sem - A pointer to the semaphore whose attributes are to be
192  * modified
193  * protocol - The new protocol to use
194  *
195  * Return Value:
196  * 0 if successful. Otherwise, -1 is returned and the errno value is set
197  * appropriately.
198  *
199  ****************************************************************************/
200 
201 int sem_setprotocol(FAR sem_t *sem, int protocol);
