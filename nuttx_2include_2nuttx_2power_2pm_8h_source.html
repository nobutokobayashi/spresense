36 /* Definition of terms. Various "sleep" and low power consumption states
37  * have various names and are sometimes used in conflicting ways. In the
38  * PM logic, we will use the following terminology:
39  *
40  * NORMAL - The normal, full power operating mode.
41  * IDLE - This is still basically normal operational mode, the system is,
42  * however, IDLE and some simple simple steps to reduce power
43  * consumption provided that they do not interfere with normal
44  * Operation. Simply dimming a backlight might be an example
45  * something that would be done when the system is idle.
46  * STANDBY - Standby is a lower power consumption mode that may involve more
47  * extensive power management steps such has disabling clocking or
48  * setting the processor into reduced power consumption modes. In
49  * this state, the system should still be able to resume normal
50  * activity almost immediately.
51  * SLEEP - The lowest power consumption mode. The most drastic power
52  * reduction measures possible should be taken in this state. It
53  * may require some time to get back to normal operation from
54  * SLEEP (some MCUs may even require going through reset).
55  *
56  * State changes always proceed from higher to lower power usage:
57  *
58  * NORMAL->IDLE->STANDBY->SLEEP
59  * ^ | | |
60  * | V V V
61  * +-------+------+--------+
62  */

117 /* The averaging algorithm is simply: Y = (An*X + SUM(Ai*Yi))/SUM(Aj), where
118  * i = 1..n-1 and j= 1..n, n is the length of the "memory", Ai is the

233 /* This enumeration provides all power management states. Receipt of the
234  * state indication is the state transition event.

237 enum pm_state_e
238 {
239  PM_NORMAL = 0, /* Normal full power operating mode. If the driver is in
240  * a reduced power usage mode, it should immediately re-
241  * initialize for normal operatin.
242  *
243  * PM_NORMAL may be followed by PM_IDLE.
244  */
245  PM_IDLE, /* Drivers will receive this state change if it is
246  * appropriate to enter a simple IDLE power state. This
247  * would include simple things such as reducing display back-
248  * lighting. The driver should be ready to resume normal
249  * activity instantly.
250  *
251  * PM_IDLE may be followed by PM_STANDBY or PM_NORMAL.
252  */
253  PM_STANDBY, /* The system is entering standby mode. Standby is a lower
254  * power consumption mode that may involve more extensive
255  * power management steps such has disabling clocking or
256  * setting the processor into reduced power consumption
257  * modes. In this state, the system should still be able
258  * to resume normal activity almost immediately.
259  *
260  * PM_STANDBY may be followed PM_SLEEP or by PM_NORMAL
261  */
262  PM_SLEEP, /* The system is entering deep sleep mode. The most drastic
263  * power reduction measures possible should be taken in this
264  * state. It may require some time to get back to normal
265  * operation from SLEEP (some MCUs may even require going
266  * through reset).
267  *
268  * PM_SLEEP may be following by PM_NORMAL
269  */
270 };

278 {
279  struct sq_entry_s entry; /* Supports a singly linked list */

281  /**************************************************************************
282  * Name: prepare
283  *
284  * Description:
285  * Request the driver to prepare for a new power state. This is a
286  * warning that the system is about to enter into a new power state. The
287  * driver should begin whatever operations that may be required to enter
288  * power state. The driver may abort the state change mode by returning
289  * a non-zero value from the callback function
290  *
291  * Input Parameters:
292  * cb - Returned to the driver. The driver version of the callback
293  * structure may include additional, driver-specific state
294  * data at the end of the structure.
295  * domain - Identifies the activity domain of the state change
296  * pmstate - Identifies the new PM state
297  *
298  * Returned Value:
299  * 0 (OK) means the event was successfully processed and that the driver
300  * is prepared for the PM state change. Non-zero means that the driver
301  * is not prepared to perform the tasks needed achieve this power setting
302  * and will cause the state change to be aborted. NOTE: The prepare
303  * method will also be recalled when reverting from lower back to higher
304  * power consumption modes (say because another driver refused a lower
305  * power state change). Drivers are not permitted to return non-zero
306  * values when reverting back to higher power consumption modes!
307  *
308  **************************************************************************/
309 
310  int (*prepare)(FAR struct pm_callback_s *cb, int domain,
311  enum pm_state_e pmstate);
312 
313  /**************************************************************************
314  * Name: notify
315  *
316  * Description:
317  * Notify the driver of new power state. This callback is called after
318  * all drivers have had the opportunity to prepare for the new power
319  * state.
320  *
321  * Input Parameters:
322  * cb - Returned to the driver. The driver version of the callback
323  * structure may include additional, driver-specific state
324  * data at the end of the structure.
325  * domain - Identifies the activity domain of the state change
326  * pmstate - Identifies the new PM state
327  *
328  * Returned Value:
329  * None. The driver already agreed to transition to the low power
330  * consumption state when when it returned OK to the prepare() call.
331  * At that time it should have made all preprations necessary to enter
332  * the new state. Now the driver must make the state transition.
333  *
334  **************************************************************************/
335 
336  void (*notify)(FAR struct pm_callback_s *cb, int domain,
337  enum pm_state_e pmstate);
338 };

395 /****************************************************************************
396  * Name: pm_activity
397  *
398  * Description:
399  * This function is called by a device driver to indicate that it is
400  * performing meaningful activities (non-idle). This increment an activity
401  * count and/or will restart a idle timer and prevent entering reduced
402  * power states.
403  *
404  * Input Parameters:
405  * domain - The domain of the PM activity
406  * priority - Activity priority, range 0-9. Larger values correspond to

412  *
413  * Returned Value:
414  * None.

416  * Assumptions:

418  * PM function that may be called from an interrupt handler!).
419  *
420  ****************************************************************************/

424 /****************************************************************************
425  * Name: pm_checkstate
426  *
427  * Description:
428  * This function is called from the MCU-specific IDLE loop to monitor the
429  * the power management conditions. This function returns the "recommended"
430  * power management state based on the PM configuration and activity
431  * reported in the last sampling periods. The power management state is
432  * not automatically changed, however. The IDLE loop must call
433  * pm_changestate() in order to make the state change.
434  *
435  * These two steps are separated because the platform-specific IDLE loop may
436  * have additional situational information that is not available to the
437  * the PM sub-system. For example, the IDLE loop may know that the
438  * battery charge level is very low and may force lower power states
439  * even if there is activity.
440  *
441  * NOTE: That these two steps are separated in time and, hence, the IDLE
442  * loop could be suspended for a long period of time between calling
443  * pm_checkstate() and pm_changestate(). The IDLE loop may need to make
444  * these calls atomic by either disabling interrupts until the state change
445  * is completed.
446  *
447  * Input Parameters:
448  * domain - The PM domain to check
449  *
450  * Returned Value:
451  * The recommended power management state.
452  *
453  ****************************************************************************/
454 
455 enum pm_state_e pm_checkstate(int domain);
456 
457 /****************************************************************************
458  * Name: pm_changestate
459  *
460  * Description:
461  * This function is used to platform-specific power management logic. It
462  * will announce the power management power management state change to all
463  * drivers that have registered for power management event callbacks.
464  *
465  * Input Parameters:
466  * domain - Identifies the domain of the new PM state
467  * newstate - Identifies the new PM state
468  *
469  * Returned Value:
470  * 0 (OK) means that the callback function for all registered drivers
471  * returned OK (meaning that they accept the state change). Non-zero
472  * means that one of the drivers refused the state change. In this case,
473  * the system will revert to the preceding state.
474  *
475  * Assumptions:
476  * It is assumed that interrupts are disabled when this function is
477  * called. This function is probably called from the IDLE loop... the
478  * lowest priority task in the system. Changing driver power management
479  * states may result in renewed system activity and, as a result, can
480  * suspend the IDLE thread before it completes the entire state change
481  * unless interrupts are disabled throughout the state change.
482  *
483  ****************************************************************************/
484 
485 int pm_changestate(int domain, enum pm_state_e newstate);
