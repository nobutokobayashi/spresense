49 /* Clocking *************************************************************************/
50 /* After power-on reset, the SAME70Q device is running out of the Master Clock using
51  * the Fast RC Oscillator running at 4 MHz.
52  *
53  * MAINOSC: Frequency = 12MHz (crystal)
54  *
55  * 300MHz Settings:
56  * PLLA: PLL Divider = 1, Multiplier = 20 to generate PLLACK = 240MHz
57  * Master Clock (MCK): Source = PLLACK, Prescalar = 1 to generate MCK = 120MHz
58  * CPU clock: 120MHz
59  *
60  * There can be two on-board crystals. However, the 32.768 crystal is not
61  * populated on the stock SAME70. The fallback is to use th on-chip, slow RC
62  * oscillator which has a frequency of 22-42 KHz, nominally 32 KHz.
63  */
64

100 /* PMC master clock register settings.
101  *
102  * BOARD_PMC_MCKR_CSS - The source of main clock input. This may be one of:
103  *
104  * PMC_MCKR_CSS_SLOW Slow Clock
105  * PMC_MCKR_CSS_MAIN Main Clock
106  * PMC_MCKR_CSS_PLLA PLLA Clock
107  * PMC_MCKR_CSS_UPLL Divided UPLL Clock
108  *
109  * BOARD_PMC_MCKR_PRES - Source clock pre-scaler. May be one of:
110  *
111  * PMC_MCKR_PRES_DIV1 Selected clock
112  * PMC_MCKR_PRES_DIV2 Selected clock divided by 2
113  * PMC_MCKR_PRES_DIV4 Selected clock divided by 4
114  * PMC_MCKR_PRES_DIV8 Selected clock divided by 8
115  * PMC_MCKR_PRES_DIV16 Selected clock divided by 16
116  * PMC_MCKR_PRES_DIV32 Selected clock divided by 32
117  * PMC_MCKR_PRES_DIV64 Selected clock divided by 64
118  * PMC_MCKR_PRES_DIV3 Selected clock divided by 3
119  *
120  * The prescaler determines (1) the CPU clock and (2) the input into the
121  * second divider that then generates the Master Clock (MCK). MCK is the
122  * source clock of the peripheral clocks.
123  *
124  * BOARD_PMC_MCKR_MDIV - MCK divider. May be one of:
125  *
126  * PMC_MCKR_MDIV_DIV1 Master Clock = Prescaler Output Clock / 1
127  * PMC_MCKR_MDIV_DIV2 Master Clock = Prescaler Output Clock / 2
128  * PMC_MCKR_MDIV_DIV4 Master Clock = Prescaler Output Clock / 4
129  * PMC_MCKR_MDIV_DIV3 Master Clock = Prescaler Output Clock / 3
130  */
131 
132 #define BOARD_PMC_MCKR_CSS PMC_MCKR_CSS_PLLA /* Source = PLLA */
133 #define BOARD_PMC_MCKR_PRES PMC_MCKR_PRES_DIV1 /* Prescaler = /1 */
134 #define BOARD_PMC_MCKR_MDIV PMC_MCKR_MDIV_DIV2 /* MCK divider = /2 */
