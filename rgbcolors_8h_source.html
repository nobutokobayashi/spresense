47 /* Color Creation and Conversion Macros *************************************/
48 /* This macro creates RGB24 from 8:8:8 RGB */
49 
50 #define RGBTO24(r,g,b) \
51  ((uint32_t)((r) & 0xff) << 16 | (uint32_t)((g) & 0xff) << 8 | (uint32_t)((b) & 0xff))
52 
53 /* And these macros perform the inverse transformation */
54 
55 #define RGB24RED(rgb) (((rgb) >> 16) & 0xff)
56 #define RGB24GREEN(rgb) (((rgb) >> 8) & 0xff)
57 #define RGB24BLUE(rgb) ( (rgb) & 0xff)
58 
59 /* This macro creates RGB16 (5:6:5) from 8:8:8 RGB:
60  *
61  * R[7:3] -> RGB[15:11]
62  * G[7:2] -> RGB[10:5]
63  * B[7:3] -> RGB[4:0]
64  */
65 
66 #define RGBTO16(r,g,b) \
67  ((((uint16_t)(r) << 8) & 0xf800) | (((uint16_t)(g) << 3) & 0x07e0) | (((uint16_t)(b) >> 3) & 0x001f))
68 
69 /* And these macros perform the inverse transformation */
70 
71 #define RGB16RED(rgb) (((rgb) >> 8) & 0xf8)
72 #define RGB16GREEN(rgb) (((rgb) >> 3) & 0xfc)
73 #define RGB16BLUE(rgb) (((rgb) << 3) & 0xf8)
74 
75 /* This macro creates RGB8 (3:3:2) from 8:8:8 RGB */
76 
77 #define RGBTO8(r,g,b) \
78  ((((uint8_t)(r) << 5) & 0xe0) | (((uint8_t)(g) << 2) & 0x1c) | ((uint8_t)(b) & 0x03))
79 
80 /* And these macros perform the inverse transformation */
81 
82 #define RGB8RED(rgb) ( (rgb) & 0xe0)
83 #define RGB8GREEN(rgb) (((rgb) << 3) & 0xe0)
84 #define RGB8BLUE(rgb) (((rgb) << 6) & 0xc0)
85 
86 /* This macro converts RGB24 (8:8:8) to RGB16 (5:6:5):
87  *
88  * 00000000 RRRRRRRR BBBBBBBB GGGGGGGG -> RRRRRBBB BBBGGGGG
89  */
90 
91 #define RGB24TO16(rgb24) \
92  (((rgb24 >> 8) & 0xf800) | ((rgb24 >> 5) & 0x07e0) | ((rgb24 >> 3) & 0x001f))
93 
94 /* This macro converts RGB16 (5:6:5) to RGB24 (8:8:8):
95  *
96  * RRRRRBBB BBBGGGGG -> 00000000 RRRRRRRR BBBBBBBB GGGGGGGG
97  */
98 
99 #define RGB16TO24(rgb16) \
100  (((rgb16 & 0xf800) << 8) | ((rgb16 & 0x07e0) << 5) | ((rgb16 & 0x001f) << 3))
