101  /* Bits 22-23: Reserved */
102 #define ADC_CR_START_SHIFT (24) /* Bits 24-26: Control A/D conversion start */
103 #define ADC_CR_START_MASK (7 << ADC_CR_START_SHIFT)
104 # define ADC_CR_START_NOSTART (0 << ADC_CR_START_SHIFT) /* No start */
105 # define ADC_CR_START_NOW (1 << ADC_CR_START_SHIFT) /* Start now */
106 # define ADC_CR_START_P2p10 (2 << ADC_CR_START_SHIFT) /* Start edge on P2.10/EINT0/NMI */
107 # define ADC_CR_START_P1p27 (3 << ADC_CR_START_SHIFT) /* Start edge on P1.27/CLKOUT/USB_OVRCRn/CAP0.1 */

109 # define ADC_CR_START_MAT0p3 (5 << ADC_CR_START_SHIFT) /* Start edge on MAT0.3 */
110 # define ADC_CR_START_MAT1p0 (6 << ADC_CR_START_SHIFT) /* Start edge on MAT1.0 */

112 #define ADC_CR_EDGE (1 << 27) /* Bit 27: Start on falling edge */
