66 /* SDIOWAIT_WRCOMPLETE (optional) : Certain SDIO driver can use D0 busy
67  * signalling to detect Write Complete. Used of D0 busy signalling will
68  * avoid potentially very long (600Ms+) busy waiting in the MMCSD driver.
69  *
70  * To implement D0 Busy signalling, the underlying SDIO driver must be
71  * capable of switching the D0 GPIO to be a rising edge sensitive
72  * interrupt pin. It must then, condition that pin to detect the rising
73  * edge on receipt of SDWAIT_WRCOMPLETE in the SDIO_WAITENABLE call and
74  * return it back to regular SDIO mode, when either the ISR fires or pin
75  * is found to be high in the SDIO_EVENTWAIT call.
76  */
77 
78 #if defined(CONFIG_MMCSD_SDIOWAIT_WRCOMPLETE)
79 # define SDIOWAIT_WRCOMPLETE (1 << 5) /* Bit 5: Hardware Write Completion */
80 # define SDIOWAIT_ALLEVENTS 0x3f

82 # define SDIOWAIT_ALLEVENTS 0x1f

90 /* Commands are bit-encoded to provide as much information to the SDIO driver as
91  * possible in 32-bits. The encoding is as follows:
92  *
93  * ---- ---- ---- ---- ---T TTRR RRCC CCCC
94  *
95  * CCCCCC - Bits 0-5: 6-bit command index (Range 9-63)
96  * RRRR - Bits 6-9: 4-bit response code (R1, R1B, R2-5)
97  * TTT - Bits 10-12: Data transfer type
98  */

217 /* Response Encodings:
218  *
219  * xxxx xxxx xxxx xxxx OSMX XXRR RRCC CCCC
220  *
221  * x - Bit not used
222  * C - Bits 0-5: Command index
223  * R - Bits 6-9: Response type
224  * X - Bits 10-12: Data transfer type
225  * M - Bit 13: Multiple block transfer
226  * S - Bit 14: Stop data transfer
227  * O - Bit 15: Open drain
