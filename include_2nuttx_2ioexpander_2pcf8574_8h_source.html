56 /* A reference to a structure of this type must be passed to the PCF8574xx
57  * driver when the driver is instantiated. This structure provides
58  * information about the configuration of the PCF8574xx and provides some
59  * board-specific hooks.
60  *
61  * Memory for this structure is provided by the caller. It is not copied by
62  * the driver and is presumed to persist while the driver is active. The
63  * memory must be writeable because, under certain circumstances, the driver
64  * may modify the frequency.
65  */
66

68 {
69  /* Device characterization */

72  uint32_t frequency; /* I2C frequency */
73 
74 #ifdef CONFIG_PCF8574_INT_ENABLE
75  /* IRQ/GPIO access callbacks. These operations all hidden behind
76  * callbacks to isolate the PCF8574xx driver from differences in GPIO
77  * interrupt handling by varying boards and MCUs.
78  *
79  * attach - Attach the PCF8574xx interrupt handler to the GPIO interrupt
80  * enable - Enable or disable the GPIO interrupt

82 
83  CODE int (*attach)(FAR struct pcf8574_config_s *state,
