69 /* The PLL generates Fvco according to the following formulae. The input clock to
70  * the PLL may be either the external crystal (Fxtal) or PIOSC (Fpiosc). This
71  * logic supports only the external crystal as the PLL source clock.
72  *
73  * Fin = Fxtal / (Q + 1 )(N + 1) -OR- Fpiosc / (Q + 1)(N + 1)
74  * Mdiv = Mint + (MFrac / 1024)
75  * Fvco = Fin * Mdiv
76  *
77  * Where the register fields Q and N actually hold (Q-1) and (N-1). The following
78  * setup then generates Fvco = 480MHz:
79  *
80  * Fin = 25 MHz / 1 / 5 = 5 MHz
81  * Mdiv = 96
82  * Fvco = 480
83  */
84 
85 #define BOARD_PLL_MINT 96 /* Integer part of PLL M value */
86 #define BOARD_PLL_MFRAC 0 /* Fractional part of PLL M value */
87 #define BOARD_PLL_N 5 /* PLL N value */
88 #define BOARD_PLL_Q 1 /* PLL Q value */
89 
90 #define BOARD_FVCO_FREQUENCY 480000000 /* Resulting Fvco */
91 
92 /* When the PLL is active, the system clock frequency (SysClk) is calculated using
93  * the following equation:
94  *
95  * SysClk = Fvco/ (sysdiv + 1)
96  *
97  * The following setup generates Sysclk = 120MHz:
98  */
99 
100 #define BOARD_PLL_SYSDIV 4 /* Sysclk = Fvco / 4 = 120MHz */
101 #define SYSCLK_FREQUENCY 120000000 /* Resulting SysClk frequency */
