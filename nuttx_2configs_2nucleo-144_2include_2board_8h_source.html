62 /* Clocking *************************************************************************/
63 /* The Nucleo-144 board provides the following clock sources:
64  *
65  * MCO: 8 MHz from MCO output of ST-LINK is used as input clock
66  * X2: 32.768 KHz crystal for LSE
67  * X3: HSE crystal oscillator (not provided)
68  *
69  * So we have these clock source available within the STM32
70  *
71  * HSI: 16 MHz RC factory-trimmed
72  * LSI: 32 KHz RC
73  * HSE: 8 MHz from MCO output of ST-LINK
74  * LSE: 32.768 kHz
75  */

84 /* Main PLL Configuration.
85  *
86  * PLL source is HSE = 8,000,000
87  *
88  * PLL_VCO = (STM32_HSE_FREQUENCY / PLLM) * PLLN
89  * Subject to:
90  *
91  * 2 <= PLLM <= 63
92  * 192 <= PLLN <= 432
93  * 192 MHz <= PLL_VCO <= 432MHz
94  *
95  * SYSCLK = PLL_VCO / PLLP
96  * Subject to
97  *
98  * PLLP = {2, 4, 6, 8}
99  * SYSCLK <= 216 MHz
100  *
101  * USB OTG FS, SDMMC and RNG Clock = PLL_VCO / PLLQ
102  * Subject to
103  * The USB OTG FS requires a 48 MHz clock to work correctly. The SDMMC
104  * and the random number generator need a frequency lower than or equal
105  * to 48 MHz to work correctly.
106  *
107  * 2 <= PLLQ <= 15
108  */
109 
110 /* Highest SYSCLK with USB OTG FS clock = 48 MHz
111  *
112  * PLL_VCO = (8,000,000 / 4) * 216 = 432 MHz
113  * SYSCLK = 432 MHz / 2 = 216 MHz
114  * USB OTG FS, SDMMC and RNG Clock = 432 MHz / 9 = 48 MHz
115  */
116
