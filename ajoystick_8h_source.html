36 /* This header file provides definition for a standard analog joystick
37  * interface. An analog joystick refers to a joystick that provides X/Y
38  * positional data as integer values such as might be provides by Analog-
39  * to-Digital Conversion (ADC). The analog positional data may also be
40  * accompanied by discrete button data.
41  *
42  * The analog joystick driver exports a standard character driver
43  * interface. By convention, the analog joystick is registered as an input
44  * device at /dev/ajoyN where N uniquely identifies the driver instance.
45  *
46  * This header file documents the generic interface that all NuttX analog
47  * joystick devices must conform. It adds standards and conventions on top
48  * of the standard character driver interface.
49  */

107 /* IOCTL commands
108  *
109  * Discrete joystick drivers do not support the character driver write() or
110  * seek() methods. The remaining driver methods behave as follows:
111  *
112  * 1) The read() method will always return a single value of size
113  * struct ajoy_sample_s represent the current joystick positional and the
114  * state of all joystick buttons. read() never blocks. X an Y position
115  * data is raw converted data. Zeroing and scaling must be performed by
116  * the application.
117  * 2) The poll() method can be used to notify a client if there is a change
118  * in any of the joystick button inputs. This feature, of course,
119  * depends upon interrupt GPIO support from the platform. NOTE: that
120  * semantics of poll() for POLLIN are atypical: The successful poll
121  * means that the button data has changed and has nothing to with the
122  * availability of data to be read; data is always available to be
123  * read.
124  * 3) The ioctl() method supports the commands documented below:
125  */
126 
127 /* Command: AJOYIOC_SUPPORTED
128  * Description: Report the set of button events supported by the hardware;
129  * Argument: A pointer to writeable integer value in which to return the
130  * set of supported buttons.
131  * Return: Zero (OK) on success. Minus one will be returned on failure
132  * with the errno value set appropriately.
133  */
134 
135 #define AJOYIOC_SUPPORTED _AJOYIOC(0x0001)
136 
137 /* Command: AJOYIOC_POLLEVENTS
138  * Description: Specify the set of button events that can cause a poll()
139  * to awaken. The default is all button depressions and all
140  * button releases (all supported buttons);
141  * Argument: A read-only pointer to an instance of struct ajoy_pollevents_s
142  * Return: Zero (OK) on success. Minus one will be returned on failure
143  * with the errno value set appropriately.
144  */
145 
146 #define AJOYIOC_POLLEVENTS _AJOYIOC(0x0002)
147 
148 /* Command: AJOYIOC_REGISTER
149  * Description: Register to receive a signal whenever there is a change in
150  * any of the joystick analog inputs. This feature, of
151  * course, depends upon interrupt GPIO support from the
152  * platform.
153  * Argument: A read-only pointer to an instance of struct ajoy_notify_s
154  * Return: Zero (OK) on success. Minus one will be returned on failure
155  * with the errno value set appropriately.
156  */
157 
158 #define AJOYIOC_REGISTER _AJOYIOC(0x0003)

217 /* The analog joystick driver is a two-part driver:
218  *
219  * 1) A common upper half driver that provides the common user interface to
220  * the joystick,
221  * 2) Platform-specific lower half drivers that provide the interface
222  * between the common upper half and the platform analog and discrete
223  * inputs.
224  *
225  * This structure defines the interface between an instance of the lower
226  * half driver and the common upper half driver. Such an instance is
227  * passed to the upper half driver when the driver is initialized, binding
228  * the upper and lower halves into one driver.
229  */
230 
231 struct ajoy_lowerhalf_s
232 {
233  /* Return the set of buttons supported on the analog joystick device */
234 
235  CODE ajoy_buttonset_t (*al_supported)(FAR const struct ajoy_lowerhalf_s *lower);
236 
237  /* Return the current state of all analog joystick position and button data */
238 
239  CODE int (*al_sample)(FAR const struct ajoy_lowerhalf_s *lower,
240  FAR struct ajoy_sample_s *sample);
241 
242  /* Return the current state of button data (only) */
243 
244  CODE ajoy_buttonset_t (*al_buttons)(FAR const struct ajoy_lowerhalf_s *lower);
245 
246  /* Enable interrupts on the selected set of joystick buttons. An empty
247  * set will disable all interrupts.
248  */
249
