51 /* ENC28J60 Commands ********************************************************/
52 /* A total of seven instructions are implemented on the ENC28J60. Where:
53  *
54  * aaaaaa is the 5-bit address of a control register, and
55  * dddddddd is one or more bytes of data that may accompany the command.
56  */
57 
58 #define ENC_RCR (0x00) /* Read Control Register
59  * 000 | aaaaa | (Register value returned)) */
60 #define ENC_RBM (0x3a) /* Read Buffer Memory
61  * 001 | 11010 | (Read buffer data follows) */
62 #define ENC_WCR (0x40) /* Write Control Register
63  * 010 | aaaaa | dddddddd */
64 #define ENC_WBM (0x7a) /* Write Buffer Memory
65  * 011 | 11010 | (Write buffer data follows) */
66 #define ENC_BFS (0x80) /* Bit Field Set
67  * 100 | aaaaa | dddddddd */
68 #define ENC_BFC (0xa0) /* Bit Field Clear
69  * 101 | aaaaa | dddddddd */
70 #define ENC_SRC (0xff) /* System Reset
71  * 111 | 11111 | (No data) */

73 /* Global Control Registers *************************************************/
74 /* Control registers are accessed with the RCR, RBM, WCR, BFS, and BFC
75  * commands. The following identifies all ENC28J60 control registers. The
76  * control register memory is partitioned into four banks, selectable by the
77  * bank select bits, BSEL1:BSEL0, in the ECON1 register.
78  *
79  * The last five locations (0x1b to 0x1f) of all banks point to a common set
80  * of registers: EIE, EIR, ESTAT, ECON2 and ECON1. These are key registers
81  * used in controlling and monitoring the operation of the device. Their
82  * common mapping allows easy access without switching the bank.
83  *
84  * Control registers for the ENC28J60 are generically grouped as ETH, MAC and
85  * MII registers. Register names starting with E belong to the ETH group.
86  * Similarly, registers names starting with MA belong to the MAC group and
87  * registers prefixed with MI belong to the MII group.
88  */
89

93 #define ENC_ECON2 (0x1e) /* Ethernet Control 2 Register */
94 #define ENC_ECON1 (0x1f) /* Ethernet Control 1 Register */
