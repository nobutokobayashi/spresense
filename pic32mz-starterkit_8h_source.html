124 /* LEDs *********************************************************************/
125 /* The PIC32 starter kit has 3 user LEDs
126  *
127  * PIN LED Notes
128  * --- ----- -------------------------
129  * RH0 LED1 High illuminates (RED)
130  * RH1 LED3 High illuminates (YELLOW)
131  * RH2 LED2 High illuminates (GREEN)
132  */
133 
134 #define GPIO_LED_1 (GPIO_OUTPUT | GPIO_VALUE_ZERO | GPIO_PORTH | GPIO_PIN0)
135 #define GPIO_LED_2 (GPIO_OUTPUT | GPIO_VALUE_ZERO | GPIO_PORTH | GPIO_PIN1)
136 #define GPIO_LED_3 (GPIO_OUTPUT | GPIO_VALUE_ZERO | GPIO_PORTH | GPIO_PIN2)
137 
138 /* The PIC32MZ Ethernet Starter kit has 3 user push buttons labelled SW1-3
139  * on the board:
140  *
141  * PIN LED Notes
142  * ---- ---- -------------------------
143  * RB12 SW1 Active-low
144  * RB13 SW2 Active-low
145  * RB14 SW3 Active-low
146  *
147  * The switches do not have any debounce circuitry and require internal pull-
148  * up resistors. When Idle, the switches are pulled high (+3.3V), and they
149  * are grounded when pressed.
150  */
151 
152 #define GPIO_SW_1 (GPIO_INPUT | GPIO_INTERRUPT | GPIO_PULLUP | \
153  GPIO_PORTB | GPIO_PIN12)
154 #define GPIO_SW_2 (GPIO_INPUT | GPIO_INTERRUPT | GPIO_PULLUP | \
155  GPIO_PORTB | GPIO_PIN13)
156 #define GPIO_SW_3 (GPIO_INPUT | GPIO_INTERRUPT | GPIO_PULLUP | \
157  GPIO_PORTB | GPIO_PIN14)
