149 #define SPI_CTRLA_SWRST (1 << 0) /* Bit 0: Software reset */
150 #define SPI_CTRLA_ENABLE (1 << 1) /* Bit 1: Enable */
151 #define SPI_CTRLA_MODE_SHIFT (2) /* Bits 2-4: Operating Mode */
152 #define SPI_CTRLA_MODE_MASK (7 << SPI_CTRLA_MODE_SHIFT)
153 # define SPI_CTRLA_MODE_SLAVE (2 << SPI_CTRLA_MODE_SHIFT) /* SPI slave operation */
154 # define SPI_CTRLA_MODE_MASTER (3 << SPI_CTRLA_MODE_SHIFT) /* SPI master operation */
155 #define SPI_CTRLA_RUNSTDBY (1 << 7) /* Bit 7: Run in standby */
156 #define SPI_CTRLA_IBON (1 << 8) /* Bit 8: Immediate BUFOVF notification */
157 #define SPI_CTRLA_DOPO_SHIFT (16) /* Bit 16-17: Data out pinout */
158 #define SPI_CTRLA_DOPO_MASK (3 << SPI_CTRLA_DOPO_SHIFT) /* Bit 16-17: Data out pinout */
159 # define SPI_CTRLA_DOPO_DOPAD012 (0 << SPI_CTRLA_DOPO_SHIFT) /* D0=PAD0 SCK=PAD1 SS=PAD2 */
160 # define SPI_CTRLA_DOPO_DOPAD231 (1 << SPI_CTRLA_DOPO_SHIFT) /* D0=PAD2 SCK=PAD3 SS=PAD1 */
161 # define SPI_CTRLA_DOPO_DOPAD312 (2 << SPI_CTRLA_DOPO_SHIFT) /* D0=PAD3 SCK=PAD1 SS=PAD2 */
162 # define SPI_CTRLA_DOPO_DOPAD031 (3 << SPI_CTRLA_DOPO_SHIFT) /* D0=PAD0 SCK=PAD3 SS=PAD1 */
163 #define SPI_CTRLA_DIPO_SHIFT (20) /* Bits 20-21: Data in pinout */
164 #define SPI_CTRLA_DIPO_MASK (3 << SPI_CTRLA_DIPO_SHIFT)
165 # define SPI_CTRLA_DIPAD0 (0 << SPI_CTRLA_DIPO_SHIFT) /* SERCOM PAD0 for DI */
166 # define SPI_CTRLA_DIPAD1 (1 << SPI_CTRLA_DIPO_SHIFT) /* SERCOM PAD1 for DI */
167 # define SPI_CTRLA_DIPAD2 (2 << SPI_CTRLA_DIPO_SHIFT) /* SERCOM PAD2 for DI */
168 # define SPI_CTRLA_DIPAD3 (3 << SPI_CTRLA_DIPO_SHIFT) /* SERCOM PAD3 for DI */
169 #define SPI_CTRLA_FORM_SHIFT (24) /* Bits 24-27: Frame format */
170 #define SPI_CTRLA_FORM_MASK (7 << SPI_CTRLA_FORM_SHIFT)
171 # define SPI_CTRLA_FORM_SPI (0 << SPI_CTRLA_FORM_SHIFT) /* SPI frame (no address) */
172 # define SPI_CTRLA_FORM_ADDR (2 << SPI_CTRLA_FORM_SHIFT) /* SPI frame (w/address) */
173 #define SPI_CTRLA_CPHA (1 << 28) /* Bit 28: Clock phase */

175 #define SPI_CTRLA_DORD (1 << 30) /* Bit 30: Data order */

177 # define SPI_CTRLA_LSBFIRST SPI_CTRLA_DORD
