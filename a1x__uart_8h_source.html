242 /* UART Interrupt Enable Register */
243 
244 #define UART_IER_ERBFI (1 << 0) /* Bit 0: Enable Received Data Available Interrupt */
245 #define UART_IER_ETBEI (1 << 1) /* Bit 1: Enable Transmit Holding Register Empty Interrupt */
246 #define UART_IER_ELSI (1 << 2) /* Bit 2: Enable Receiver Line Status Interrupt */
247 #define UART_IER_EDSSI (1 << 3) /* Bit 3: Enable Modem Status Interrupt */

251 /* UART Interrupt Identity Register */
252 
253 #define UART_IIR_IID_SHIFT (0) /* Bits: 0-3: Interrupt ID */
254 #define UART_IIR_IID_MASK (15 << UART_IIR_IID_SHIFT)
255 # define UART_IIR_IID_MODEM (0 << UART_IIR_IID_SHIFT) /* Modem status */
256 # define UART_IIR_IID_NONE (1 << UART_IIR_IID_SHIFT) /* No interrupt pending */
257 # define UART_IIR_IID_TXEMPTY (2 << UART_IIR_IID_SHIFT) /* THR empty */
258 # define UART_IIR_IID_RECV (4 << UART_IIR_IID_SHIFT) /* Received data available */
259 # define UART_IIR_IID_LINESTATUS (6 << UART_IIR_IID_SHIFT) /* Receiver line status */
260 # define UART_IIR_IID_BUSY (7 << UART_IIR_IID_SHIFT) /* Busy detect */
261 # define UART_IIR_IID_TIMEOUT (12 << UART_IIR_IID_SHIFT) /* Character timeout */
262 #define UART_IIR_FEFLAG_SHIFT (6) /* Bits 6-7: FIFOs Enable Flag */
263 #define UART_IIR_FEFLAG_MASK (3 << UART_IIR_FEFLAG_SHIFT)
264 # define UART_IIR_FEFLAG_DISABLE (0 << UART_IIR_FEFLAG_SHIFT)
265 # define UART_IIR_FEFLAG_ENABLE (3 << UART_IIR_FEFLAG_SHIFT)
266 
267 /* UART FIFO Control Register */
268 
269 #define UART_FCR_FIFOE (1 << 0) /* Bit 0: Enable FIFOs */
270 #define UART_FCR_RFIFOR (1 << 1) /* Bit 1: RCVR FIFO Reset */
271 #define UART_FCR_XFIFOR (1 << 2) /* Bit 2: XMIT FIFO reset */
272 #define UART_FCR_DMAM (1 << 3) /* Bit 3: DMA mode */
273 #define UART_FCR_TFT_SHIFT (4) /* Bits 4-5: TX Empty Trigger */
274 #define UART_FCR_TFT_MASK (3 << UART_FCR_TFT_SHIFT)
275 # define UART_FCR_TFT_EMPTY (0 << UART_FCR_TFT_SHIFT) /* FIFO empty */
276 # define UART_FCR_TFT_TWO (1 << UART_FCR_TFT_SHIFT) /* 2 characters in the FIFO */
277 # define UART_FCR_TFT_QUARTER (2 << UART_FCR_TFT_SHIFT) /* FIFO 1/4 full */
278 # define UART_FCR_TFT_HALF (3 << UART_FCR_TFT_SHIFT) /* FIFO 1/2 full */
279 #define UART_FCR_RT_SHIFT (6) /* Bits 6-7: RCVR Trigger */
280 #define UART_FCR_RT_MASK (3 << UART_FCR_RT_SHIFT)
281 # define UART_FCR_RT_ONE (0 << UART_FCR_RT_SHIFT) /* 1 character in the FIFO */
282 # define UART_FCR_RT_QUARTER (1 << UART_FCR_RT_SHIFT) /* FIFO 1/4 full */
283 # define UART_FCR_RT_HALF (2 << UART_FCR_RT_SHIFT) /* FIFO 1/2 full */
284 # define UART_FCR_RT_MINUS2 (3 << UART_FCR_RT_SHIFT) /* FIFO-2 less than full */
