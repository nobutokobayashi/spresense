54 /* Clock Sources
55  * - 1-28 MHz High Frequency RC Oscillator (HFRCO)
56  * - 4-32 MHz High Frequency Crystal Oscillator (HFXO)

73 /* HFCLK - High Frequency Clock
74  *
75  * HFCLK is the selected High Frequency Clock. This clock is used by the CMU
76  * and drives the two prescalers that generate HFCORECLK and HFPERCLK. The
77  * HFCLK can be driven by a high-frequency oscillator (HFRCO or HFXO) or one
78  * of the low-frequency oscillators (LFRCO or LFXO). By default the HFRCO is
79  * selected.
80  */
81 
82 #define BOARD_HFCLKSEL _CMU_CMD_HFCLKSEL_HFXO
83 #define BOARD_HFCLKDIV 0 /* Does not apply to EFM32G */
84 #define BOARD_HFCLK_FREQUENCY BOARD_HFXO_FREQUENCY
85 
86 /* HFCORECLK - High Frequency Core Clock
87  *
88  * HFCORECLK is a prescaled version of HFCLK. This clock drives the Core
89  * Modules, which consists of the CPU and modules that are tightly coupled
90  * to the CPU, e.g. MSC, DMA etc. The frequency of HFCORECLK is set using
91  * the CMU_HFCORECLKDIV register.
92  */
93 
94 #define BOARD_HFCORECLKDIV _CMU_HFCORECLKDIV_HFCORECLKDIV_DEFAULT
95 #define BOARD_HFCORECLK_FREQUENCY BOARD_HFXO_FREQUENCY
96 
97 /* HFPERCLK - High Frequency Peripheral Clock
98  *
99  * Like HFCORECLK, HFPERCLK can also be a prescaled version of HFCLK. This
100  * clock drives the High-Frequency Peripherals. The frequency of HFPERCLK is
101  * set using the CMU_HFPERCLKDIV register.
102  */
103 
104 #define BOARD_HFPERCLKDIV _CMU_HFPERCLKDIV_HFPERCLKDIV_DEFAULT

107 /* LFACLK - Low Frequency A Clock
108  *
109  * LFACLK is the selected clock for the Low Energy A Peripherals. There are
110  * four selectable sources for LFACLK: LFRCO, LFXO, HFCORECLK/2 and ULFRCO.
111  * From reset, the LFACLK source is set to LFRCO. However, note that the
112  * LFRCO is disabled from reset. The selection is configured using the LFA
113  * field in CMU_LFCLKSEL. The HFCORECLK/2 setting allows the Low Energy A
114  * Peripherals to be used as high-frequency peripherals.
115  *
116  * Use _CMU_LFCLKSEL_LFA_DISABLED to disable
117  * ULFRCO is a special case.
118  */
119 
120 #define BOARD_LFACLKSEL _CMU_LFCLKSEL_LFA_LFXO
121 #undef BOARD_LFACLK_ULFRCO
122 #define BOARD_LFACLK_FREQUENCY BOARD_LFXO_FREQUENCY
123 
124 /* LFBCLK - Low Frequency B Clock
125  *
126  * LFBCLK is the selected clock for the Low Energy B Peripherals. There are
127  * four selectable sources for LFBCLK: LFRCO, LFXO, HFCORECLK/2 and ULFRCO.
128  * From reset, the LFBCLK source is set to LFRCO. However, note that the
129  * LFRCO is disabled from reset. The selection is configured using the LFB
130  * field in CMU_LFCLKSEL. The HFCORECLK/2 setting allows the Low Energy B
131  * Peripherals to be used as high-frequency peripherals.
132  *
133  * Use _CMU_LFCLKSEL_LFB_DISABLED to disable
134  * ULFRCO is a special case.
135  */
136

141 /* PCNTnCLK - Pulse Counter n Clock
142  *
143  * Each available pulse counter is driven by its own clock, PCNTnCLK where
144  * n is the pulse counter instance number. Each pulse counter can be
145  * configured to use an external pin (PCNTn_S0) or LFACLK as PCNTnCLK.
146  */

148 /* WDOGCLK - Watchdog Timer Clock
149  *
150  * The Watchdog Timer (WDOG) can be configured to use one of three different
151  * clock sources: LFRCO, LFXO or ULFRCO. ULFRCO (Ultra Low Frequency RC
152  * Oscillator) is a separate 1 kHz RC oscillator that also runs in EM3.
153  */
154 
155 /* AUXCLK - Auxiliary Clock
156  *
157  * AUXCLK is a 1-28 MHz clock driven by a separate RC oscillator, AUXHFRCO.
158  * This clock is used for flash programming and Serial Wire Output (SWO).
159  * During flash programming this clock will be active. If the AUXHFRCO has
160  * not been enabled explicitly by software, the MSC will automatically
161  * start and stop it. The AUXHFRCO is enabled by writing a 1 to AUXHFRCOEN
162  * in CMU_OSCENCMD. This explicit enabling is required when SWO is used.
163  */
