54 template<typename T, /* element type */
55  typename NumT = size_t> /* element num type */
56 class RuntimeQue : CopyGuard {
57  void post_push() { ++m_count; if (++m_put == capacity()) m_put = 0; }
58  void post_pop() { --m_count; if (++m_get == capacity()) m_get = 0; }
59 
60  NumT get_index(NumT n) const {
61  return static_cast<NumT>((m_get + n < capacity()) ? m_get + n : m_get + n - capacity());
62  }
63 
64 public:
65  RuntimeQue(void* area, NumT depth) :
66  m_data(static_cast<T*>(area)),
67  m_capacity(depth),
68  m_put(0), m_get(0), m_count(0) {
69 // D_ASSERT(que_area && depth);
70  }
71  ~RuntimeQue() { clear(); }
72 
73  const void* que_area() const { return m_data; }
74  size_t elem_size() const { return sizeof(T); }
75  NumT capacity() const { return m_capacity; }
76  NumT size() const { return m_count; }
77  NumT rest() const { return static_cast<NumT>(capacity() - size()); }
78  bool empty() const { return size() == 0; }
79  bool full() const { return size() == capacity(); }
80 
81  void dump() const {
82  printf("dump: put=%u, get=%u, capa=%u, size=%u, rest=%u, empty=%d, full=%d\n",
83  m_put, m_get, capacity(), size(), rest(), empty(), full());
84  for (NumT i = 0; i < size(); ++i) {
85  printf("%u: data=%08x\n",

87  }
88  }

91  while (pop()) ;
92  m_put = m_get = 0;
93  }

95  /* キュー末尾にデータを入れる(placement newによるコピーコンストラクタ使用) */
96  bool push(const T& data) {
97  if (full()) return false;
98  ::new(&m_data[m_put]) T(data);
99  DBG_P("push: pos=%u, cnt=%u\n", m_put, m_count);
100  post_push();
101  return true;
102  }
103 
104  /* キュー先頭のデータをデストラクタ呼出し後に取り除く */
105  bool pop() {
106  if (empty()) return false;
107  DBG_P("pop : pos=%u, cnt=%u\n", m_get, m_count);
108  m_data[m_get].~T();
109  post_pop();
110  return true;
111  }

114  const T& top() const { return at(0); }

116  /* キューのN番目(先頭が0)のデータを参照する */
117  const T& at(NumT n) const {
118  D_ASSERT(n < m_count); /* 例外は使えないので、assertとする */
119  DBG_P("at(%u): pos=%u\n", n, m_get);
120  return m_data[get_index(n)];
121  }
