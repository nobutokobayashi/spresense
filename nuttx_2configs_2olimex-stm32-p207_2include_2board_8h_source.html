70 /* Main PLL Configuration.
71  *
72  * PLL source is HSE
73  * PLL_VCO = (STM32_HSE_FREQUENCY / PLLM) * PLLN
74  * = (25,000,000 / 25) * 240
75  * = 240,000,000
76  * SYSCLK = PLL_VCO / PLLP
77  * = 240,000,000 / 2 = 120,000,000
78  * USB OTG FS, SDIO and RNG Clock

80  * = 240,000,000 / 5 = 48,000,000
81  * = 48,000,000
82  */
83
