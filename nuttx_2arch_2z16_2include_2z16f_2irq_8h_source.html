96 /* IRQ Stack Frame Format
97  *
98  * This stack frame is created on each interrupt. These registers are stored
99  * in the TCB to many context switches.
100  *
101  * The following represent all of the "static" registers r8-r15. These
102  * are registers that whose value must be retained across function calls.
103  * These registers must be saved bothby interrupt handling context switch
104  * switch logic and also by user-initiated context switches.
105  *
106  * Registers are saved in the order consistent with pushmho <r8-r15>,
107  * that is with r15 pushed first and r8 push last. Since the z16f has
108  * a "push-down" stack, the registers will be "in order" in memory.
109  */
110 
111 #define REG_R8 ( 0) /* 32-bits: R8 */
112 #define REG_R9 ( 2) /* 32-bits: R9 */
113 #define REG_R10 ( 4) /* 32-bits: R10 */
114 #define REG_R11 ( 6) /* 32-bits: R11 */
115 #define REG_R12 ( 8) /* 32-bits: R12 */
116 #define REG_R13 (10) /* 32-bits: R13 */
117 
118 /* The frame pointer and the SP at the point of task resumption must
119  * always be saved.
120  */
121 
122 #define REG_R14 (12) /* 32-bits: R14 = fp */
123 #define REG_FP REG_R14
124 #define REG_R15 (14) /* 32-bits: R15 = sp */
125 #define REG_SP REG_R15

127 /* The following represent all of the "volatile" registers r0-r7. These
128  * are registers that whose value need not be retained across function
129  * calls. These registers must be saved by interrupt handling context
130  * switch logic but not by user-initiated context switches.
131  *
132  * Registers are saved in the order consistent with pushmlo <r0-r7>,
133  * that is with r7 pushed first and r0 push last. Since the z16f has
134  * a "push-down" stack, the registers will be "in order" in memory.
135  */
