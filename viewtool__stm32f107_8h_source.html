174 /* USB
175  *
176  * The Viewtool base board has a USB Mini-B connector. Only USB device can
177  * be supported with this connector.
178  *
179  * ------------------------- ------------------------------------
180  * USB Connector
181  * J10 mini-USB GPIO CONFIGURATION(s)
182  * --- --------- ----------- ------------------------------------
183  * Pin Signal
184  * --- --------- ----------- ------------------------------------
185  * 1 USB_VBUS VDD_USB (No sensing available)
186  * 2 OTG_DM PA11 GPIO_OTGFS_DM (F107) GPIO_USB_DM (F103)
187  * 3 OTG_DP PA12 GPIO_OTGFS_DP (F107) GPIO_USB_DP (F103)
188  * 4 OTG_ID PA10 GPIO_OTGFS_ID (F107)
189  * 5 Shield N/A N/A
190  * 6 Shield N/A N/A
191  * 7 Shield N/A N/A
192  * 8 Shield N/A N/A
193  * 9 Shield N/A N/A
194  * PE11 USB_EN GPIO controlled soft pull-up (if J51 closed)
195  *
196  * NOTES:
197  * 1. GPIO_OTGFS_VBUS (F107) should not be configured. No VBUS sensing
198  * 2. GPIO_OTGFS_SOF (F107) is not used
199  * 3. The OTG FS module has is own, internal soft pull-up logic. J51 should
200  * be open so that PE11 activity does effect USB.
201  */
202

208 /* LCD
209  *
210  * An LCD may be connected via J11. Only the STM32F103 supports the FSMC signals
211  * needed to drive the LCD.
212  *
213  * The LCD features an (1) HY32D module with built-in SSD1289 LCD controller, and (a)
214  * a XPT2046 touch screen controller.
215  *
216  * LCD Connector
217  * -------------
218  *
219  * ----------------------------- ------------------------ --------------------------------
220  * Connector J11 GPIO CONFIGURATION(s)
221  * PIN SIGNAL LEGEND (F103 only) LCD Module
222  * --- ------------- ----------- ------------------------ --------------------------------
223  * 1 VDD_5 NC N/A 5V ---
224  * 2 GND GND N/A GND ---
225  * 3 PD14 DATA0 GPIO_NPS_D0 D0 HY32D
226  * 4 PD15 DATA1 GPIO_NPS_D1 D1 HY32D
227  * 5 PD0 DATA2 GPIO_NPS_D2 D2 HY32D
228  * 6 PD1 DATA3 GPIO_NPS_D3 D3 HY32D
229  * 7 PE7 DATA4 GPIO_NPS_D4 D4 HY32D
230  * 8 PE8 DATA5 GPIO_NPS_D5 D5 HY32D
231  * 9 PE9 DATA6 GPIO_NPS_D6 D6 HY32D
232  * 10 PE10 DATA7 GPIO_NPS_D7 D7 HY32D
233  * 11 PE11 DATA8 GPIO_NPS_D8 D8 HY32D
234  * 12 PE12 DATA9 GPIO_NPS_D9 D9 HY32D
235  * 13 PE13 DATA10 GPIO_NPS_D10 D10 HY32D
236  * 14 PE14 DATA11 GPIO_NPS_D11 D11 HY32D
237  * 15 PE15 DATA12 GPIO_NPS_D12 D12 HY32D
238  * 16 PD8 DATA13 GPIO_NPS_D13 D13 HY32D
239  * 17 PD9 DATA14 GPIO_NPS_D14 D14 HY32D
240  * 18 PD10 DATA15 GPIO_NPS_D15 D15 HY32D
241  * 19 (3) LCD_CS GPIO_NPS_NE1 CS HY32D
242  * 20 PD11 LCD_RS GPIO_NPS_A16 RS HY32D
243  * 21 PD5 LCD_R/W GPIO_NPS_NWE WR HY32D
244  * 22 PD4 LCD_RD GPIO_NPS_NOE RD HY32D
245  * 23 PB1 LCD_RESET (GPIO) RESET HY32D
246  * 24 N/C NC N/A TE (unused?)
247  * 25 VDD_3.3 BL_VCC N/A BLVDD CA6219 (Drives LCD backlight)
248  * 26 GND BL_GND N/A BLGND CA6219
249  * 27 PB0 BL_PWM GPIO_TIM3_CH3OUT(2) BL_CNT CA6219
250  * 28 PC5 LCDTP_IRQ (GPIO) TP_IRQ XPT2046
251  * 29 PC4 LCDTP_CS (GPIO) TP_CS XPT2046
252  * 30 PB13 LCDTP_CLK GPIO_SPI2_SCK TP_SCK XPT2046
253  * 31 PB15 LCDTP_DIN GPIO_SPI2_MOSI TP_SI XPT2046
254  * 32 PB14 LCDTP_DOUT GPIO_SPI2_MISO TP_SO XPT2046
255  * 33 VDD_3.3 VDD_3.3 N/A 3.3V ---
256  * 34 GND GND N/A GND ---
257  * --- ------------- ----------- ------------------------ --------------------------------
258  *
259  * NOTES:
260  * 1) Only the F103 version of the board supports the FSMC
261  * 2) No remap
262  * 3) LCD_CS is controlled by J13 JUMPER4 (under the LCD unfortunately):
263  *
264  * 1->2 : PD7 (GPIO_NPS_NE1) enables the multiplexor : 1E\ enable input (active LOW)
265  * 3->4 : PD13 provides 1A0 input (1A1 is grounded). : 1A0 address input
266  * So will chip enable to either LCD_CS or
267  * Flash_CS.
268  * 5->6 : 1Y0 output to LCD_CS : 1Y0 address output
269  * 7->8 : 1Y1 output to Flash_CE : 1Y1 address output
270  *
271  * Truth Table:
272  * 1E\ 1A0 1A1 1Y0 1Y1
273  * --- --- --- --- ---
274  * HI N/A N/A HI HI
275  * LO LO LO LO HI
276  * LO HI LO HI LO
277  */
278
