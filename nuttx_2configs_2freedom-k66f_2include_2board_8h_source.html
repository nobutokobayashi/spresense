55 /* Clocking *************************************************************************/
56 /* The Freedom K66F uses a 12Mhz external Oscillator. The Kinetis MCU startup from an
57  * internal digitally-controlled oscillator (DCO). Nuttx will enable the main external
58  * oscillator (EXTAL0/XTAL0). The external oscillator/resonator can range from
59  * 32.768 KHz up to 50 MHz. The default external source for the MCG oscillator inputs
60  * is 12 MHz oscillator
61  *
62  * X501 a High-frequency, low-power Xtal
63  */
64 
65 #define BOARD_EXTAL_LP 1
66 #define BOARD_EXTAL_FREQ 12000000 /* 12MHz Oscillator */
67 #define BOARD_XTAL32_FREQ 32768 /* 32KHz RTC Oscillator */
68 
69 /* PLL Configuration. Either the external clock or crystal frequency is used to
70  * select the PRDIV value. Only reference clock frequencies are supported that will
71  * produce a KINETIS_MCG_PLL_REF_MIN >= PLLIN <=KINETIS_MCG_PLL_REF_MAX reference
72  * clock to the PLL.
73  *
74  * PLL Input frequency: PLLIN = REFCLK / PRDIV = 12 Mhz / 1 = 12 MHz
75  * PLL Output frequency: PLLOUT = PLLIN * VDIV = 12 Mhz * 30 = 360 MHz
76  * MCG Frequency: PLLOUT = 180 Mhz = 360 MHz / KINETIS_MCG_PLL_INTERNAL_DIVBY
77  *
78  * PRDIV register value is the divider minus KINETIS_MCG_C5_PRDIV_BASE.
79  * VDIV register value is offset by KINETIS_MCG_C6_VDIV_BASE.
80  */
81 
82 #define BOARD_PRDIV 1 /* PLL External Reference Divider */
83 #define BOARD_VDIV 30 /* PLL VCO Divider (frequency multiplier) */
84 
85 /* Define additional MCG_C2 Setting */

293 /* An alternative serial port might use a standard serial shield mounted
294  * on the Freedom Board. In this case, Arduino pin D1 provides UART TX and
295  * pin D0 privies UART RX.
296  *
297  * The I/O headers on the FRDM-K66F board are arranged to enable
298  * compatibility with Arduino shield. The outer rows of pins (even numbered
299  * pins) on the headers, share the same mechanical spacing and placement with
300  * the I/O headers on the Arduino Revision 3 (R3) standard.

302  * The Arduino D0 and D1 pins then correspond to pins 2 and 4 on the J1 I/O
303  * connector:
304  *
305  * Arduino Pin FRDM-K66F J1 Connector
306  * ------------------------ -----------------------
307  * UART RX, Arduino D0 pin Pin 2, PTC3, UART1_RX
308  * UART TX, Arduino D1 pin Pin 4, PTC4, UART1_TX
