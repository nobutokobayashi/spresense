318 /* The TIVA/Stellaris only supports 8 priority levels. The hardware priority
319  * mechanism will only look at the upper N bits of the 8-bit priority level
320  * (where N is 3 for the Tiva/Stellaris family), so any prioritization must be
321  * performed in those bits. The default priority level is set to the middle
322  * value
323  */
324 
325 #define NVIC_SYSH_PRIORITY_MIN 0xe0 /* Bits [7:5] set in minimum priority */
326 #define NVIC_SYSH_PRIORITY_DEFAULT 0x80 /* Midpoint is the default */
327 #define NVIC_SYSH_PRIORITY_MAX 0x00 /* Zero is maximum priority */
328 #define NVIC_SYSH_PRIORITY_STEP 0x20 /* Three bits of interrupt priority used */
329 
330 /* If CONFIG_ARMV7M_USEBASEPRI is selected, then interrupts will be disabled
331  * by setting the BASEPRI register to NVIC_SYSH_DISABLE_PRIORITY so that most
332  * interrupts will not have execution priority. SVCall must have execution
333  * priority in all cases.
334  *
335  * In the normal cases, interrupts are not nest-able and all interrupts run
336  * at an execution priority between NVIC_SYSH_PRIORITY_MIN and
337  * NVIC_SYSH_PRIORITY_MAX (with NVIC_SYSH_PRIORITY_MAX reserved for SVCall).
338  *
339  * If, in addition, CONFIG_ARCH_HIPRI_INTERRUPT is defined, then special
340  * high priority interrupts are supported. These are not "nested" in the
341  * normal sense of the word. These high priority interrupts can interrupt
342  * normal processing but execute outside of OS (although they can "get back
343  * into the game" via a PendSV interrupt).
344  *
345  * In the normal course of things, interrupts must occasionally be disabled
346  * using the up_irq_save() inline function to prevent contention in use of
347  * resources that may be shared between interrupt level and non-interrupt
348  * level logic. Now the question arises, if CONFIG_ARCH_HIPRI_INTERRUPT,
349  * do we disable all interrupts (except SVCall), or do we only disable the
350  * "normal" interrupts. Since the high priority interrupts cannot interact
351  * with the OS, you may want to permit the high priority interrupts even if
352  * interrupts are disabled. The setting CONFIG_ARCH_INT_DISABLEALL can be
353  * used to select either behavior:
354  *
355  * ----------------------------+--------------+----------------------------
356  * CONFIG_ARCH_HIPRI_INTERRUPT | NO | YES
357  * ----------------------------+--------------+--------------+-------------
358  * CONFIG_ARCH_INT_DISABLEALL | N/A | YES | NO
359  * ----------------------------+--------------+--------------+-------------
360  * | | | SVCall
361  * | SVCall | SVCall | HIGH
362  * Disable here and below --------> MAXNORMAL ---> HIGH --------> MAXNORMAL
363  * | | MAXNORMAL |
364  * ----------------------------+--------------+--------------+-------------
365  */
366 
367 #if defined(CONFIG_ARCH_HIPRI_INTERRUPT) && defined(CONFIG_ARCH_INT_DISABLEALL)
368 # define NVIC_SYSH_MAXNORMAL_PRIORITY (NVIC_SYSH_PRIORITY_MAX + 2*NVIC_SYSH_PRIORITY_STEP)
369 # define NVIC_SYSH_HIGH_PRIORITY (NVIC_SYSH_PRIORITY_MAX + NVIC_SYSH_PRIORITY_STEP)
370 # define NVIC_SYSH_DISABLE_PRIORITY NVIC_SYSH_HIGH_PRIORITY
371 # define NVIC_SYSH_SVCALL_PRIORITY NVIC_SYSH_PRIORITY_MAX
372 #else
373 # define NVIC_SYSH_MAXNORMAL_PRIORITY (NVIC_SYSH_PRIORITY_MAX + NVIC_SYSH_PRIORITY_STEP)
374 # define NVIC_SYSH_HIGH_PRIORITY NVIC_SYSH_PRIORITY_MAX
375 # define NVIC_SYSH_DISABLE_PRIORITY NVIC_SYSH_MAXNORMAL_PRIORITY
376 # define NVIC_SYSH_SVCALL_PRIORITY NVIC_SYSH_PRIORITY_MAX
377 #endif
