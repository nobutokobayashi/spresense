56 /* Clocking *************************************************************************/
57 /* Overview
58  *
59  * Since there is not high speed crystal, we will run from the OSC16M clock source.
60  * We will use its default, POR frequency of 4MHz to avoid an additional clock
61  * switch.
62  *
63  * OSC16M Output = 4MHz
64  * `- GCLK1 Input = 4MHz Prescaler = 1 output = 4MHz
65  * `- DFLL Input = 4MHz Multiplier = 12 output = 48MHz
66  * `- GCLK0 Input = 48MHz Prescaler = 1 output = 48MHz
67  * `- MCLK Input = 48Mhz CPU divider = 1 CPU frequency = 48MHz
68  * APBA divider = 1 APBA frequency = 48MHz
69  * APBB divider = 1 APBB frequency = 48MHz
70  * APBC divider = 1 APBC frequency = 48MHz
71  * APBD divider = 1 APBD frequency = 48MHz
72  * APBE divider = 1 APBE frequency = 48MHz
73  *
74  * The SAML21 Xplained Pro has one on-board crystal:
75  *
76  * XC101 32.768KHz XOSC32
77  *
78  * REVISIT: Not currently used, may want to use as GCLK1 source with
79  * DFLL multiplier of ((48000000+16384)/32768) = 1465 which would yield
80  * a clock of 48,005,120 MHz.
81  */
82

190 /* Digital Frequency Locked Loop configuration. In closed-loop mode, the
191  * DFLL output frequency (Fdfll) is given by:
192  *
193  * Fdfll = DFLLmul * Frefclk
194  * = 12 * 4000000 = 48MHz
195  *
196  * Where the reference clock is Generic Clock Channel 0 output of GLCK1.
197  * GCLCK1 provides OSC16M, undivided.
198  *
199  * When operating in open-loop mode, the output frequency of the DFLL will
200  * be determined by the values written to the DFLL Coarse Value bit group
201  * and the DFLL Fine Value bit group in the DFLL Value register.
202  *
203  * BOARD_DFLL48M_CLOSEDLOOP - Boolean (defined / not defined)
204  * BOARD_DFLL48M_OPENLOOP - Boolean (defined / not defined)
205  * BOARD_DFLL48M_RECOVERY - Boolean (defined / not defined)
206  * BOARD_DFLL48M_TRACKAFTERFINELOCK - Boolean (defined / not defined)
207  * BOARD_DFLL48M_KEEPLOCKONWAKEUP - Boolean (defined / not defined)
208  * BOARD_DFLL48M_ENABLECHILLCYCLE - Boolean (defined / not defined)
209  * BOARD_DFLL48M_QUICKLOCK - Boolean (defined / not defined)
210  * BOARD_DFLL48M_RUNINSTDBY - Boolean (defined / not defined)
211  * BOARD_DFLL48M_ONDEMAND - Boolean (defined / not defined)
212  * BOARD_DFLL48M_COARSEVALUE - Value
213  * BOARD_DFLL48M_FINEVALUE - Value
214  *
215  * Open Loop mode only:
216  * BOARD_DFLL48M_COARSEVALUE - Value
217  * BOARD_DFLL48M_FINEVALUE - Value
218  *
219  * Closed loop mode only:
220  * BOARD_DFLL48M_REFCLK_CLKGEN - GCLK index in the range {0..8}
221  * BOARD_DFLL48M_MULTIPLIER - Value
222  * BOARD_DFLL48M_MAXCOARSESTEP - Value
223  * BOARD_DFLL48M_MAXFINESTEP - Value
224  *
225  * BOARD_DFLL48M_FREQUENCY - The resulting frequency
226  */
