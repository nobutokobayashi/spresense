53 /* Clocking *************************************************************************/
54 /* The Kwikstik-K40 has a 4MHz crystal on board */
55 
56 #undef BOARD_EXTCLOCK /* Crystal */
57 #define BOARD_EXTAL_FREQ 4000000 /* 4MHz crystal frequency (REFCLK) */
58 #define BOARD_XTAL32_FREQ 32768 /* 32KHz RTC Oscillator */
59 
60 /* PLL Configuration. NOTE: Only even frequency crystals are supported that will
61  * produce a 2MHz reference clock to the PLL.
62  *
63  * PLL Input frequency: PLLIN = REFCLK/PRDIV = 4MHz/2 = 2MHz
64  * PLL Output frequency: PLLOUT = PLLIN*VDIV = 2Mhz*48 = 96MHz
65  * MCG Frequency: PLLOUT = 96MHz
66  */
67 
68 #define BOARD_PRDIV 2 /* PLL External Reference Divider */
69 #define BOARD_VDIV 48 /* PLL VCO Divider (frequency multiplier) */
70 
71 #define BOARD_PLLIN_FREQ (BOARD_EXTAL_FREQ / BOARD_PRDIV)
72 #define BOARD_PLLOUT_FREQ (BOARD_PLLIN_FREQ * BOARD_VDIV)
73 #define BOARD_MCG_FREQ BOARD_PLLOUT_FREQ
74 
75 /* SIM CLKDIV1 dividers */

78 #define BOARD_OUTDIV2 2 /* Bus = MCG/2, 48MHz */

80 #define BOARD_OUTDIV4 4 /* Flash clock = MCG/4, 24MHz */
81

89 /* SDCLK configurations corresponding to various modes of operation. Formula is:
90  *
91  * SDCLK frequency = (base clock) / (prescaler * divisor)

93  * The SDHC module is always configure configured so that the core clock is the base
94  * clock.
