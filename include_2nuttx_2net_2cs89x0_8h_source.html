52 /* This structure encapsulates all state information for a single hardware
53  * interface. It includes values that must be provided by the user to in
54  * to describe details of the CS89x00 implementation on a particular board.
55  * An instance if this structure is passed to cs89x00 to instantiate the
56  * driver.
57  *
58  * This structure also includes internal driver state information that should
59  * be of no concern to the caller of cs89x0_initialize(). These fields must
60  * be zeroed.
61  */
62 
63 struct cs89x0_driver_s
64 {
65  /* User-provided CS89x00 platform-specific implementation details. The
66  * caller of cs89x0_initialize() must provide all of these values.
67  */
68 
69  FAR void *cs_base; /* CS89x0 region base address */
70 #ifdef CONFIG_CS89x0_MEMMODE
71  FAR void *cs_ppbase; /* CS89x0 page packet base address */
72 #endif
73  uint8_t cs_irq; /* CS89x00 IRQ number */
74 
75  /* Driver internal state fields. These must be zeroed by before the
76  * instance of this structure is passed to cs89x0_initialize
77  */
78 #ifdef CONFIG_CS89x0_XMITEARLY
79  uint8_t txstart; /* Bits 6-7 of TxCMD controls Tx race */
80 #endif
81  bool cs_memmode; /* true:memory mode false: I/O mode */
82  bool cs_bifup; /* true:ifup false:ifdown */
83  WDOG_ID cs_txpoll; /* TX poll timer */
84  WDOG_ID cs_txtimeout; /* TX timeout timer */
85 #ifdef CONFIG_CS89x0_XMITEARLY
86  uint32_t cs_txunderrun; /* Count of Tx underrun errors */
87 #endif
88 
89  /* This holds the information visible to the NuttX network */
90 
91  struct net_driver_s cs_dev; /* Interface understood by the network */
92 };
