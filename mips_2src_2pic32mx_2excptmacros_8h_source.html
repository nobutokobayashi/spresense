82 /********************************************************************************************
83  * Name: EXCPT_PROLOGUE
84  *
85  * Description:
86  * Provides the "prologue" logic that should appear at the beginning of every exception
87  * handler.
88  *
89  * On Entry:
90  * sp - Points to the top of the stack
91  * tmp - Is a register the can be modified for scratch usage (after it has been saved)
92  * k0 and k1 - Since we are in an exception handler, these are available for use
93  *
94  * At completion:
95  * Register state is saved on the stack; All registers are available for usage except sp
96  * and k1:
97  *
98  * - sp points the beginning of the register save area
99  * - k1 holds the value of the STATUS register
100  *
101  * The following registers are modified: k0, k1, sp, a0
102  *
103  ********************************************************************************************/

363 /********************************************************************************************
364  * Name: USE_INTSTACK
365  *
366  * Description:
367  * Switch to the interrupt stack (if enabled in the configuration).
368  *
369  * On Entry:
370  * sp - Current value of the user stack pointer
371  * tmp1, tmp2, tmp3, and tmp4 are registers that can be used temporarily.
372  * All interrupts should still be disabled.
373  *
374  * At completion:
375  * If the nesting level is 0, then (1) the user stack pointer is saved at the base of the
376  * interrupt stack and sp points to the interrupt stack.
377  * The values of tmp1, tmp2, tmp3, and sp have been altered
378  *
379  ********************************************************************************************/
380 
381  .macro USE_INTSTACK, tmp1, tmp2, tmp3, tmp4
382 
383 #if CONFIG_ARCH_INTERRUPTSTACK > 3
384 #ifdef CONFIG_PIC32MX_NESTED_INTERRUPTS
385 
386  /* Check the nesting level. If there are no nested interrupts, then we can
387  * claim the interrupt stack.
388  */
389 
390  la \tmp1, g_nestlevel
391  lw \tmp2, (\tmp1)
392  bne 1f
393  nop
394 #endif
395 
396  /* Use the interrupt stack, pushing the user stack pointer onto the interrupt
397  * stack first.
398  */
399 
400  la \tmp3, g_intstackbase
401  lw \tmp4, (\tmp3)
402  sw sp, (\tmp4)
403  move sp, \tmp4
404 
405 #ifdef CONFIG_PIC32MX_NESTED_INTERRUPTS
406 1:
407  /* Increment the interrupt nesting level */
408 
409  addiu \tmp2, \tmp2, 1
410  sw \tmp2, 0(\tmp1)
411 #endif
412 #endif
413  .endm

415 /********************************************************************************************
416  * Name: RESTORE_STACK
417  *
418  * Description:
419  * Restore the user stack. Not really.. actually only decrements the nesting level. We
420  * always get the new stack pointer for the register save array.
421  *
422  * On Entry:
423  * tmp1 and tmp2 are registers that can be used temporarily.
424  * All interrupts must be disabled.
425  *
426  * At completion:
427  * Current nesting level is decremented
428  * The values of tmp1 and tmp2 have been altered
429  *
430  ********************************************************************************************/
431 
432  .macro RESTORE_STACK, tmp1, tmp2
433 
434 #if CONFIG_ARCH_INTERRUPTSTACK > 3

436 
437  /* Decrement the nesting level */
438 
439  la \tmp1, g_nestlevel
440  lw \tmp2, (\tmp1)
441  addiu \tmp2, \tmp2, -1
442  sw \tmp2, 0(\tmp1)
443 
444 #endif
