106 /* LED definitions ******************************************************************/
107 /* There is an RGB LED on board the SAMA5D2-XULT. The RED component is driven by
108  * the SDHC_CD pin (PA13) and so will not be used. The LEDs are provided VDD_LED
109  * and so bringing the LED low will will illuminated the LED.
110  *
111  * ------------------------------ ------------------- -------------------------
112  * SAMA5D2 PIO SIGNAL USAGE
113  * ------------------------------ ------------------- -------------------------
114  * PA13 SDHC_CD_PA13 Red LED
115  * PB5 LED_GREEN_PB5 Green LED
116  * PB0 LED_BLUE_PB0 Blue LED
117  * ------------------------------ ------------------- -------------------------
118  */

143 /* These LEDs are not used by the board port unless CONFIG_ARCH_LEDS is
144  * defined. In that case, the usage by the board port is defined in
145  * include/board.h and src/sam_leds.c. The LEDs are used to encode OS-related
146  * events as follows. Note that only the GREEN LED is used in this case
147  *
148  * SYMBOL Val Meaning Green LED
149  * ----------------- --- ----------------------- ----------- */
150 #define LED_STARTED 0 /* NuttX has been started OFF */
151 #define LED_HEAPALLOCATE 0 /* Heap has been allocated OFF */
152 #define LED_IRQSENABLED 0 /* Interrupts enabled OFF */
153 #define LED_STACKCREATED 1 /* Idle stack created ON */
154 #define LED_INIRQ 2 /* In an interrupt N/C */
155 #define LED_SIGNAL 2 /* In a signal handler N/C */
156 #define LED_ASSERTION 2 /* An assertion failed N/C */
157 #define LED_PANIC 3 /* The system has crashed Flash */
158 #undef LED_IDLE /* MCU is is sleep mode Not used */
159 
160 /* Thus if the Green LED is statically on, NuttX has successfully booted and
161  * is, apparently, running normally. If LED is flashing at approximately
162  * 2Hz, then a fatal error has been detected and the system has halted.
