54 /* Clocking *************************************************************************/
55 /* Four different clock sources can be used to drive the system clock (SYSCLK):
56  *
57  * - HSI high-speed internal oscillator clock
58  * Generated from an internal 8 MHz RC oscillator
59  * - HSE high-speed external oscillator clock
60  * Normally driven by an external crystal (X3). However, this crystal is not
61  * fitted on the Nucleo-F072RB board.
62  * - PLL clock
63  * - MSI multispeed internal oscillator clock
64  * The MSI clock signal is generated from an internal RC oscillator. Seven frequency
65  * ranges are available: 65.536 kHz, 131.072 kHz, 262.144 kHz, 524.288 kHz, 1.048 MHz,
66  * 2.097 MHz (default value) and 4.194 MHz.
67  *
68  * The devices have the following two secondary clock sources
69  * - LSI low-speed internal RC clock
70  * Drives the watchdog and RTC. Approximately 37KHz
71  * - LSE low-speed external oscillator clock
72  * Driven by 32.768KHz crystal (X2) on the OSC32_IN and OSC32_OUT pins.
73  */
74

84 /* PLL Configuration
85  *
86  * - PLL source is HSI -> 8MHz input (nominal)
87  * - PLL source predivider 2 -> 4MHz divided down PLL VCO clock output
88  * - PLL multipler is 12 -> 48MHz PLL VCO clock output (for USB)
89  *
90  * Resulting SYSCLK frequency is 8MHz x 12 / 2 = 48MHz
91  *
92  * USB:
93  * If the USB interface is used in the application, it requires a precise
94  * 48MHz clock which can be generated from either the (1) the internal
95  * main PLL with the HSE clock source using an HSE crystal oscillator. In
96  * this case, the PLL VCO clock (defined by STM32F0_CFGR_PLLMUL) must be
97  * programmed to output a 96 MHz frequency. This is required to provide a
98  * 48MHz clock to the (USBCLK = PLLVCO/2). Or (2) by using the internal
99  * 48MHz oscillator in automatic trimming mode. The synchronization for
100  * this oscillator can be taken from the USB data stream itself (SOF
101  * signalization) which allows crystal-less operation.
102  * SYSCLK
103  * The system clock is derived from the PLL VCO divided by the output
104  * division factor.
105  * Limitations:
106  * - 96 MHz as PLLVCO when the product is in range 1 (1.8V),
107  * - 48 MHz as PLLVCO when the product is in range 2 (1.5V),
108  * - 24 MHz when the product is in range 3 (1.2V).
109  * - Output division to avoid exceeding 32 MHz as SYSCLK.
110  * - The minimum input clock frequency for PLL is 2 MHz (when using HSE as
111  * PLL source).
112  */
113

187 /* These LEDs are not used by the board port unless CONFIG_ARCH_LEDS is
188  * defined. In that case, the usage by the board port is defined in
189  * include/board.h and src/sam_leds.c. The LEDs are used to encode OS-related
190  * events as follows when the red LED (PE24) is available:
191  *
192  * SYMBOL Meaning LD2
193  * ------------------- ----------------------- -----------
194  * LED_STARTED NuttX has been started OFF
195  * LED_HEAPALLOCATE Heap has been allocated OFF
196  * LED_IRQSENABLED Interrupts enabled OFF
197  * LED_STACKCREATED Idle stack created ON
198  * LED_INIRQ In an interrupt No change
199  * LED_SIGNAL In a signal handler No change
200  * LED_ASSERTION An assertion failed No change
201  * LED_PANIC The system has crashed Blinking
202  * LED_IDLE MCU is is sleep mode Not used
203  *
204  * Thus if LD2, NuttX has successfully booted and is, apparently, running
205  * normally. If LD2 is flashing at approximately 2Hz, then a fatal error
206  * has been detected and the system has halted.
207  */
208

210 #define LED_HEAPALLOCATE 0
211 #define LED_IRQSENABLED 0
212 #define LED_STACKCREATED 1
213 #define LED_INIRQ 2
214 #define LED_SIGNAL 2
215 #define LED_ASSERTION 2
216 #define LED_PANIC 1
