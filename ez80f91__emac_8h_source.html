207 #define EMAC_RXDESC_OVR 0x0001 /* Bit 0: 1=A Receive Overrun occurs in this packet. An
208  * overrun occurs when all of the EMAC Receive
209  * buffers are in use and the Receive FIFO is full.
210  * The hardware ignores all incoming packets until the
211  * ISTAT Register [RXOVR] bit is cleared by the software.
212  * There is no indication as to how many packets are
213  * ignored. */

215  * Indicates that the last Receive event is not long

218  * defined as Rx error RxER = 1, receive data valid
219  * (RxDV) = 0 and receive data (RxD) = Eh */
220 #define EMAC_RXDESC_CODEV 0x0008 /* Bit 3: 1=A code violation is detected. The PHY asserts
221  * Rx error (RxER). */
222 #define EMAC_RXDESC_LCERROR 0x0010 /* Bit 4 1=Type/Length field is not a Type field and it does
223  * not match the actual data byte length of the
224  * Ethernet packet. The data byte length is the
225  * number of bytes of data in the Ethernet packet
226  * between the Type/Length field and the FCS. */
227 #define EMAC_RXDESC_LOOR 0x0020 /* Bit 5: 1=Type/Length field is out of range (larger
