81 /* Detailed contents of save areas.
82  * NOTE: caller must define the XCHAL_SA_REG macro (not defined here)
83  * before expanding the XCHAL_xxx_SA_LIST() macros.
84  *
85  * XCHAL_SA_REG(s,ccused,abikind,kind,opt,name,galign,align,asize,
86  * dbnum,base,regnum,bitsz,gapsz,reset,x...)
87  *
88  * s = passed from XCHAL_*_LIST(s), eg. to select how to expand
89  * ccused = set if used by compiler without special options or code
90  * abikind = 0 (caller-saved), 1 (callee-saved), or 2 (thread-global)
91  * kind = 0 (special reg), 1 (TIE user reg), or 2 (TIE regfile reg)
92  * opt = 0 (custom TIE extension or coprocessor), or 1 (optional reg)
93  * name = lowercase reg name (no quotes)
94  * galign = group byte alignment (power of 2) (galign >= align)
95  * align = register byte alignment (power of 2)
96  * asize = allocated size in bytes (asize*8 == bitsz + gapsz + padsz)
97  * (not including any pad bytes required to galign this or next reg)
98  * dbnum = unique target number f/debug (see <xtensa-libdb-macros.h>)
99  * base = reg shortname w/o index (or sr=special, ur=TIE user reg)
100  * regnum = reg index in regfile, or special/TIE-user reg number
101  * bitsz = number of significant bits (regfile width, or ur/sr mask bits)
102  * gapsz = intervening bits, if bitsz bits not stored contiguously
103  * (padsz = pad bits at end [TIE regfile] or at msbits [ur,sr] of asize)
104  * reset = register reset value (or 0 if undefined at reset)
105  * x = reserved for future use (0 until then)
106  *
107  * To filter out certain registers, e.g. to expand only the non-global
108  * registers used by the compiler, you can do something like this:
109  *
110  * #define XCHAL_SA_REG(s,ccused,p...) SELCC##ccused(p)
111  * #define SELCC0(p...)
112  * #define SELCC1(abikind,p...) SELAK##abikind(p)
113  * #define SELAK0(p...) REG(p)
114  * #define SELAK1(p...) REG(p)
115  * #define SELAK2(p...)
116  * #define REG(kind,tie,name,galn,aln,asz,csz,dbnum,base,rnum,bsz,rst,x...) \
117  * ...what you want to expand...
118  */
