150 /* SSI Control 0 (SSICR0), offset 0x000 */
151 
152 #define SSI_CR0_DSS_SHIFT 0 /* Bits 3-0: SSI Data Size Select */
153 #define SSI_CR0_DSS_MASK (0x0f << SSI_CR0_DSS_SHIFT)
154 #define SSI_CR0_DSS(n) ((n-1) << SSI_CR0_DSS_SHIFT) /* n={4,5,..16} */
155 #define SSI_CR0_FRF_SHIFT 4 /* Bits 5-4: SSI Frame Format Select */
156 #define SSI_CR0_FRF_MASK (3 << SSI_CR0_FRF_SHIFT)
157 #define SSI_CR0_FRF_SPI (0 << SSI_CR0_FRF_SHIFT) /* Freescale SPI format */
158 #define SSI_CR0_FRF_SSFF (1 << SSI_CR0_FRF_SHIFT) /* TI synchronous serial fram format */
159 #define SSI_CR0_FRF_UWIRE (2 << SSI_CR0_FRF_SHIFT) /* MICROWIRE frame format */
160  #define SSI_CR0_SPO (1 << 6) /* Bit 6: SSI Serial Clock Polarity */
161 #define SSI_CR0_SPH (1 << 7) /* Bit 7: SSI Serial Clock Phase */
162 #define SSI_CR0_SCR_SHIFT 8 /* Bits 15-8: SSI Serial Clock Rate */
163 #define SSI_CR0_SCR_MASK (0xff << SSI_CR0_SCR_SHIFT)
164 
165 /* SSI Control 1 (SSICR1), offset 0x004 */
166 
167 #define SSI_CR1_LBM (1 << 0) /* Bit 0: SSI Loopback Mode */
168 #define SSI_CR1_SSE (1 << 1) /* Bit 1: SSI Synchronous Serial Port Enable */
169 #define SSI_CR1_MS (1 << 2) /* Bit 2: SSI Master/Slave Select slave */
170 #define SSI_CR1_SOD (1 << 3) /* Bit 3: SSI Slave Mode Output Disable */
