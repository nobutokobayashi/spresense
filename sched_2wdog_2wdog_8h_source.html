108 /****************************************************************************
109  * Name: wd_timer
110  *
111  * Description:
112  * This function is called from the timer interrupt handler to determine
113  * if it is time to execute a watchdog function. If so, the watchdog
114  * function will be executed in the context of the timer interrupt
115  * handler.
116  *
117  * Parameters:
118  * ticks - If CONFIG_SCHED_TICKLESS is defined then the number of ticks
119  * in the interval that just expired is provided. Otherwise,
120  * this function is called on each timer interrupt and a value of one
121  * is implicit.
122  *
123  * Return Value:
124  * If CONFIG_SCHED_TICKLESS is defined then the number of ticks for the
125  * next delay is provided (zero if no delay). Otherwise, this function
126  * has no returned value.
127  *
128  * Assumptions:
129  * Called from interrupt handler logic with interrupts disabled.
130  *
131  ****************************************************************************/
132 
133 #ifdef CONFIG_SCHED_TICKLESS
134 unsigned int wd_timer(int ticks);
