48 /* How can we access the errno variable? */
49 
50 #if !defined(CONFIG_BUILD_PROTECTED) && !defined(CONFIG_BUILD_KERNEL)
51  /* Flat build */
52 
53 # if defined(CONFIG_LIB_SYSCALL) && !defined(__KERNEL__)
54  /* We still might be using system calls in user code. If so, then
55  * user code will have no direct access to the errno variable.
56  */
57 
58 # undef __DIRECT_ERRNO_ACCESS
59 
60 # else
61  /* Flat build with no system calls OR internal kernel logic... There
62  * is direct access.
63  */
64 
65 # define __DIRECT_ERRNO_ACCESS 1
66 # endif
67 
68 #elif defined(CONFIG_BUILD_PROTECTED)
69 # if defined(__KERNEL__)
70  /* Kernel portion of protected build. Kernel code has direct access */
71 
72 # define __DIRECT_ERRNO_ACCESS 1
73 
74 # else
75  /* User portion of protected build. Application code has only indirect
76  * access
77  */
78 
79 # undef __DIRECT_ERRNO_ACCESS
80 # endif
81 
82 #elif defined(CONFIG_BUILD_KERNEL) && !defined(__KERNEL__)
83 # if defined(__KERNEL__)
84  /* Kernel build. Kernel code has direct access */
85 
86 # define __DIRECT_ERRNO_ACCESS 1
87 
88 # else
89  /* User libraries for the kernel. Only indirect access from user
90  * libraries */
91 
92 # undef __DIRECT_ERRNO_ACCESS
93 # endif
94 #endif

96 /* Convenience/compatibility definition.
97  *
98  * For a flat, all kernel-mode build, the error can be read and written
99  * from all code using a simple pointer.
100  */
101 
102 #ifdef __DIRECT_ERRNO_ACCESS
103 
104 # define errno *get_errno_ptr()
105 # define set_errno(e) do { errno = (int)(e); } while (0)
106 # define get_errno(e) errno
107 
108 #else
109 
110 /* We doing separate user-/kernel-mode builds, then the errno has to be
111  * a little differently. In kernel-mode, the TCB errno value can still be
112  * read and written using a pointer from code executing within the
113  * kernel.
114  *
115  * But in user-mode, the errno can only be read using the name 'errno'.
116  * The non-standard API set_errno() must explicitly be used from user-
117  * mode code in order to set the errno value.
118  *
119  * The same is true of the case where we have syscalls enabled but this
120  * is not a kernel build, then we really have no option but to use the
121  * set_errno() accessor function explicitly, even from OS logic!
122  */
123 
124 # define errno get_errno()
125 
126 #endif /* __DIRECT_ERRNO_ACCESS */
