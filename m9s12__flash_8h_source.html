50 /* Flash memory map *****************************************************************
51  *
52  * 0x4000-0x7fff: 16Kb Fixed FLASH EEPROM (Page 3e)
53  * 0x8000-0xbfff: 16Kb Page window
54  * 0xc000-0xffff: 16Kb Fixed FLASH EEPROM (Page 3f)
55  * (see chip.h)
56  *
57  * The M9S12 implements 6 bits of the PPAGE register which gives it a
58  * 1 Mbyte program memory address space that is accessed through the PPAGE
59  * window. The lower 768K portion of the address space, accessed with PPAGE
60  * values $00 through $2F, is reserved for external memory when the part is
61  * operated in expanded mode. The upper 256K of the address space, accessed
62  * with PPAGE values $30 through $3F.
63  */
64

67 /* User-Accessible utility subroutines provided by the serial monitor:
68  *
69  * PutChar - Sends the character in A out SCI0
70  *
71  * PutChar: brclr SCI0SR1,TDRE,PutChar ;wait for Tx ready
72  * staa SCI0DRL ;send character from A
73  * rts
74  *
75  * GetChar - Wait indefinitely for a character to be received via SCI.
76  * Return received character in A.
77  *
78  * GetChar: brset SCI0SR1,RDRF,RxReady ;exit loop when RDRF=1
79  * bra GetChar ;loop till RDRF set
80  * RxReady: ldaa SCI0DRL ;read character into A
81  * rts ;return
82  * EraseAllCmd - Use repeated page erase commands to erase all flash
83  * except bootloader in protected block at the end of flash, and mass
84  * erase all EEPROM locations
85  * DoOnStack - Copy to stack and execute from RAM
86  * WriteD2IX - Write the data in D (word) to the address in IX
87  * The location may be RAM, FLASH, EEPROM, or a register
88  * if FLASH or EEPROM, the operation is completed before return
89  * IX and A preserved, returns Z=1 (.EQ.) if OK
90  */
91
