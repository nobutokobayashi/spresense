318 /****************************************************************************
319  * Name: exepath_init
320  *
321  * Description:
322  * Initialize for the traversal of each value in the PATH variable. The
323  * usage is sequence is as follows:
324  *
325  * 1) Call exepath_init() to initialize for the traversal. exepath_init()
326  * will return an opaque handle that can then be provided to
327  * exepath_next() and exepath_release().
328  * 2) Call exepath_next() repeatedly to examine every file that lies
329  * in the directories of the PATH variable
330  * 3) Call exepath_release() to free resources set aside by exepath_init().
331  *
332  * Input Parameters:
333  * None
334  *
335  * Returned Value:
336  * On success, exepath_init() return a non-NULL, opaque handle that may
337  * subsequently be used in calls to exepath_next() and exepath_release().
338  * On error, a NULL handle value will be returned. The most likely cause
339  * of an error would be that there is no value associated with the PATH
340  * variable.
341  *
342  ****************************************************************************/
343 
344 #if !defined(CONFIG_BINFMT_DISABLE) && defined(CONFIG_BINFMT_EXEPATH)
345 EXEPATH_HANDLE exepath_init(void);
346 #endif

348 /****************************************************************************
349  * Name: exepath_next
350  *
351  * Description:
352  * Traverse all possible values in the PATH variable in attempt to find
353  * the full path to an executable file when only a relative path is
354  * provided.
355  *
356  * Input Parameters:
357  * handle - The handle value returned by exepath_init
358  * relpath - The relative path to the file to be found.
359  *
360  * Returned Value:
361  * On success, a non-NULL pointer to a null-terminated string is provided.
362  * This is the full path to a file that exists in the file system. This
363  * function will verify that the file exists (but will not verify that it
364  * is marked executable).
365  *
366  * NOTE: The string pointer return in the success case points to allocated
367  * memory. This memory must be freed by the called by calling kmm_free().
368  *
369  * NULL is returned if no path is found to any file with the provided
370  * 'relpath' from any absolute path in the PATH variable. In this case,
371  * there is no point in calling exepath_next() further; exepath_release()
372  * must be called to release resources set aside by expath_init().
373  *
374  ****************************************************************************/
375 
376 #if !defined(CONFIG_BINFMT_DISABLE) && defined(CONFIG_BINFMT_EXEPATH)
377 FAR char *exepath_next(EXEPATH_HANDLE handle, FAR const char *relpath);
378 #endif
