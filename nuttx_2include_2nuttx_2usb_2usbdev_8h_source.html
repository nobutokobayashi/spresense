395 /****************************************************************************
396  * Name: usbdev_dma_alloc and usbdev_dma_free
397  *
398  * Description:
399  * The USB class driver allocates packet I/O buffers for data transfer by
400  * calling the driver allocbuffer() and freebuffer() methods. Those
401  * methods are only available if CONFIG_USBDEV_DMA is defined in the
402  * system configuration.
403  *
404  * If CONFIG_USBDEV_DMAMEMORY is also defined in the NuttX configuration,
405  * then the driver implementations of the allocbuffer() and freebuffer()
406  * methods may use board-specific usbdev_dma_alloc() and usbdev_dma_free().
407  * If CONFIG_USBDEV_DMA and CONFIG_USBDEV_DMAMEMORY are both defined,
408  * then the board-specific logic must provide the functions
409  * usbdev_dma_alloc() and usbdev_dma_free() as prototyped below:
410  * usbdev_dma_alloc() will allocate DMA-capable memory of the specified
411  * size; usbdev_dma_free() is the corresponding function that will be
412  * called to free the DMA-capable memory.
413  *
414  * This functions may be simple wrappers around gran_alloc() and
415  * gran_free() (See nuttx/mm/gran.h). Note that the gran_free() function
416  * does require the size of the allocation to be freed; that would need
417  * to be managed in the board-specific logic.
418  *
419  ****************************************************************************/
420 
421 #if defined(CONFIG_USBDEV_DMA) && defined(CONFIG_USBDEV_DMAMEMORY)
422 FAR void *usbdev_dma_alloc(size_t size);
423 void usbdev_dma_free(FAR void *memory);
