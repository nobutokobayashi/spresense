124 /* Descriptor Memory Layout *********************************************************/
125 /* EMAC DMA RAM and descriptor definitions. The configured number of descriptors
126  * will determine the organization and the size of the descriptor and status tables.
127  * There is a complex interaction between the maximum packet size (CONFIG_NET_ETH_MTU)
128  * and the number of Rx and Tx descriptors that can be supported (CONFIG_NET_NRXDESC
129  * and CONFIG_NET_NTXDESC): Small buffers -> more packets. This is something that
130  * needs to be tuned for you system.
131  *
132  * For a 16Kb SRAM region, here is the relationship:
133  *
134  * 16384 <= ntx * (pktsize + 8 + 4) + nrx * (pktsize + 8 + 8)
135  *
136  * If ntx == nrx and pktsize == 590+2, then you could have ntx = nrx = 13. In this
137  * case, you would need only 15,756 bytes of EMAC RAM (but be careful with alignment!
138  * 15,756 is not well aligned.).
139  */
140 
141 #define LPC17_TXDESCTAB_SIZE (CONFIG_NET_NTXDESC*LPC17_TXDESC_SIZE)
142 #define LPC17_TXSTATTAB_SIZE (CONFIG_NET_NTXDESC*LPC17_TXSTAT_SIZE)
143 #define LPC17_TXTAB_SIZE (LPC17_TXDESCTAB_SIZE+LPC17_TXSTATTAB_SIZE)
144 
145 #define LPC17_RXDESCTAB_SIZE (CONFIG_NET_NRXDESC*LPC17_RXDESC_SIZE)
146 #define LPC17_RXSTATTAB_SIZE (CONFIG_NET_NRXDESC*LPC17_RXSTAT_SIZE)
147 #define LPC17_RXTAB_SIZE (LPC17_RXDESCTAB_SIZE+LPC17_RXSTATTAB_SIZE)
148 
149 #define LPC17_DESCTAB_SIZE (LPC17_TXTAB_SIZE+LPC17_RXTAB_SIZE)
