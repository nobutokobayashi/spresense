109 /* SPIRIT Status. This definition represents the single field of the SPIRIT
110  * status returned on each SPI transaction, equal also to the MC_STATE registers.
111  * This field-oriented structure allows user to address in simple way the single
112  * field of the SPIRIT status.
113  * The user shall define a variable of SpiritStatus type to access on SPIRIT status fields.
114  * NOTE: The fields order in the structure depends on used endianness (little or big
115  * endian). The actual definition is valid ONLY for LITTLE ENDIAN mode. Be sure to
116  * change opportunely the fields order when use a different endianness.
117  */
118 
119 #ifndef CONFIG_ENDIAN_BIG
120 struct spirit_status_s
121 {
122  uint8_t XO_ON : 1; /* Notifies if XO is operating (XO_ON is
123  * 1) or not (XO_On is 0) */
124  uint8_t MC_STATE : 7; /* Indicates the state of the Main Controller
125  * of SPIRIT. The possible states and their
126  * corresponding values are defined in enum
127  * spirit_state_e */
128  uint8_t ERROR_LOCK : 1; /* Notifies if there is an error on RCO
129  * calibration (ERROR_LOCK is 1) or not
130  * (ERROR_LOCK is 0) */
131  uint8_t RX_FIFO_EMPTY : 1; /* Notifies if RX FIFO is empty (RX_FIFO_EMPTY
132  * is 1) or not (RX_FIFO_EMPTY is 0) */
133  uint8_t TX_FIFO_FULL : 1; /* Notifies if TX FIFO is full (TX_FIFO_FULL
134  * is 1) or not (TX_FIFO_FULL is 0) */
135  uint8_t ANT_SELECT : 1; /* Notifies the currently selected antenna */
136  uint8_t reserved : 4; /* Reserved and equal to 5 */
137 };
138 #endif

140 /* One instance of this structure represents the overall state of one Spirit

142  * supported by the library with multiple instances of this structure.
