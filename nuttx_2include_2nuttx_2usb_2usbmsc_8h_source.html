91 /****************************************************************************
92  * Name: usbmsc_configure
93  *
94  * Description:
95  * One-time initialization of the USB storage driver. The initialization
96  * sequence is as follows:
97  *
98  * 1. Call usbmsc_configure to perform one-time initialization specifying
99  * the number of luns.
100  * 2. Call usbmsc_bindlun to configure each supported LUN
101  * 3. Call usbmsc_exportluns when all LUNs are configured
102  *
103  * Input Parameters:
104  * nluns - the number of LUNs that will be registered
105  * handle - Location to return a handle that is used in other API calls.
106  *
107  * Returned Value:
108  * 0 on success; a negated errno on failure. The returned handle value is
109  * an untyped equivalent to the usbmsc_classobject().
110  *
111  ****************************************************************************/
112 
113 int usbmsc_configure(unsigned int nluns, FAR void **handle);
114 
115 /****************************************************************************
116  * Name: usbmsc_bindlun
117  *
118  * Description:
119  * Bind the block driver specified by drvrpath to a USB storage LUN.
120  *
121  * Input Parameters:
122  * handle - The handle returned by a previous call to
123  * usbmsc_configure().
124  * drvrpath - the full path to the block driver
125  * startsector - A sector offset into the block driver to the start of the
126  * partition on drvrpath (0 if no partitions)
127  * nsectors - The number of sectors in the partition (if 0, all sectors
128  * to the end of the media will be exported).
129  * lunno - the LUN to bind to
130  *
131  * Returned Value:
132  * 0 on success; a negated errno on failure.
133  *
134  ****************************************************************************/
135 
136 int usbmsc_bindlun(FAR void *handle, FAR const char *drvrpath, unsigned int lunno,
137  off_t startsector, size_t nsectors, bool readonly);
138
