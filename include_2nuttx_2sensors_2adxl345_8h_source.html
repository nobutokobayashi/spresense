288 typedef void (*adxl345_handler_t)(FAR struct adxl345_config_s *config, FAR void *arg);

290 /* A reference to a structure of this type must be passed to the ADXL345 driver when the
291  * driver is instantiated. This structure provides information about the configuration of the
292  * ADXL345 and provides some board-specific hooks.
293  *
294  * Memory for this structure is provided by the caller. It is not copied by the driver
295  * and is presumed to persist while the driver is active. The memory must be writeable
296  * because, under certain circumstances, the driver may modify the frequency.
297  */
298 
299 struct adxl345_config_s
300 {
301  /* Device characterization */
302 
303 #ifdef CONFIG_ADXL345_I2C
304  uint8_t address; /* 7-bit I2C address (only bits 0-6 used) */
305 #endif
306  uint32_t frequency; /* I2C or SPI frequency */
307 
308  /* IRQ/GPIO access callbacks. These operations all hidden behind
309  * callbacks to isolate the ADXL345 driver from differences in GPIO
310  * interrupt handling by varying boards and MCUs.
311  *
312  * attach - Attach the ADXL345 interrupt handler to the GPIO interrupt
313  * enable - Enable or disable the GPIO interrupt
314  * clear - Acknowledge/clear any pending GPIO interrupt
315  */
316 
317  int (*attach)(FAR struct adxl345_config_s *state, adxl345_handler_t handler,
318  FAR void *arg);
319  void (*enable)(FAR struct adxl345_config_s *state, bool enable);
320  void (*clear)(FAR struct adxl345_config_s *state);
321 };
