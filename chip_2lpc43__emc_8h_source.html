208 /* Register Bit Definitions *************************************************************************/
209 
210 /* EMC Control register */
211 
212 #define EMC_CONTROL_ENA (1 << 0) /* Bit 0: EMC Enable */
213 #define EMC_CONTROL_ADDRMIRROR (1 << 1) /* Bit 1: Address mirror */
214 #define EMC_CONTROL_LOWPOWER (1 << 2) /* Bit 2: Low-power mode */
215  /* Bits 3-31: Reserved */
216 /* EMC Status register */
217 
218 #define EMC__
219 #define EMC_STATUS_BUSY (1 << 0) /* Bit 0: Busy */
220 #define EMC_STATUS_WB (1 << 1) /* Bit 1: Write buffer status */
221 #define EMC_CONFIG_SA (1 << 2) /* Bit 2: Self-refresh acknowledge */
222  /* Bits 3-31: Reserved */
223 /* EMC Configuration register */
224 
225 #define EMC_CONFIG_EM (1 << 0) /* Bit 0: Endian mode */
226  /* Bits 1-7: Reserved */
227 #define EMC_CONFIG_CR (1 << 8) /* Bit 8: Clock Ratio */
228  /* Bits 9-31: Reserved */
229 /* Dynamic Memory Control register */
230 
231 #define EMC_DYNCONTROL_
232 #define EMC_DYNCONTROL_CE (1 << 0) /* Bit 0: Dynamic memory clock enable */
233 #define EMC_DYNCONTROL_CS (1 << 1) /* Bit 1: Dynamic memory clock control */
234 #define EMC_DYNCONTROL_SR (1 << 2) /* Bit 2: Self-refresh request, EMCSREFREQ */
235  /* Bits 3-4: Reserved */
236 #define EMC_DYNCONTROL_MMC (1 << 5) /* Bit 5: Memory clock control */
237  /* Bit 6: Reserved */
238 #define EMC_DYNCONTROL_SI_SHIFT (7) /* Bits 7-8: SDRAM initialization */
239 #define EMC_DYNCONTROL_SI_MASK (3 << EMC_DYNCONTROL_SI_SHIFT)
240 # define EMC_DYNCONTROL_SI_NORMAL (0 << EMC_DYNCONTROL_SI_SHIFT) /* SDRAM NORMAL command */
241 # define EMC_DYNCONTROL_SI_MODE (1 << EMC_DYNCONTROL_SI_SHIFT) /* SDRAM MODE command */
242 # define EMC_DYNCONTROL_SI_PALL (2 << EMC_DYNCONTROL_SI_SHIFT) /* SDRAM PALL (precharge all) command */
243 # define EMC_DYNCONTROL_SI_ MASK (3 << EMC_DYNCONTROL_SI_SHIFT) /* SDRAM NOP (no operation) command) */
244  /* Bits 9-31: Reserved */
245 /* Dynamic Memory Refresh Timer register */
246 
247 #define EMC_DYNREFRESH_SHIFT (0) /* Bits 0-10: Refresh timer */
248 #define EMC_DYNREFRESH_MASK (0x7ff << EMC_DYNREFRESH_SHIFT)
249  /* Bits 11-31: Reserved */
250 /* Dynamic Memory Read Configuration register */
251 
252 #define EMC_DYNREADCONFIG_SHIFT (0) /* Bits 0-1: Read data strategy */
253 #define EMC_DYNREADCONFIG_MASK (3 << EMC_DYNREADCONFIG_SHIFT)
254 # define EMC_DYNREADCONFIG_0p5CCLK (1 << EMC_DYNREADCONFIG_SHIFT) /* Command delayed by 0.5 CCLK */
255 # define EMC_DYNREADCONFIG_1p5CCLK (2 << EMC_DYNREADCONFIG_SHIFT) /* Command delayed by 1.5 CCLK */
256 # define EMC_DYNREADCONFIG_2p5CCLK (3 << EMC_DYNREADCONFIG_SHIFT) /* Command delayed by 2.5 CCLK */
257  /* Bits 2-31: Reserved */
258 /* Dynamic Memory Precharge Command Period register */
259 
260 #define EMC_DYNRP_SHIFT (0) /* Bits 0-3: Precharge command period */
261 #define EMC_DYNRP_MASK (15 << EMC_DYNRP_SHIFT)
262 # define EMC_DYNRP(n) (((n)-1) << EMC_DYNRP_SHIFT) /* Delay n CCLK cycles */
263  /* Bits 2-31: Reserved */
264 /* Dynamic Memory Active to Precharge Command Period register */
265 
266 #define EMC_DYNRAS_SHIFT (0) /* Bits 0-3: Active to precharge command period */
267 #define EMC_DYNRAS_MASK (15 << EMC_DYNRAS_SHIFT)
268 # define EMC_DYNRAS(n) (((n)-1) << EMC_DYNRAS_SHIFT) /* Delay n CCLK cycles */
269  /* Bits 4-31: Reserved */
270 /* Dynamic Memory Self Refresh Exit Time register */
271 
272 #define EMC_DYNSREX_SHIFT (0) /* Bits 0-3: Self-refresh exit time */
273 #define EMC_DYNSREX_MASK (15 << EMC_DYNSREX_SHIFT)
274 # define EMC_DYNSREX(n) (((n)-1) << EMC_DYNSREX_SHIFT) /* Delay n CCLK cycles */
275  /* Bits 4-31: Reserved */
