95 /* PLL1 Setup:
96  *
97  * PLL1 input clock: CLK2 = Main OSC = 4MHz
98  * PLL1 output clock: PLL1OUT = 16 * CLK2 / 2 = 32MHz
99  * PLL1 output: CLK3 = PLL1OUT = 32MHz (hard coded selection)
100  * RCLK = CLK3 = 32MHz (hard coded selection)
101  * APB1 peripheral clock: PCLK1 = RCLK = 32MHz
102  * APB2 peripheral clock: PCLK2 = RCLK = 32MHz
103  * Main system clock: MCLK = RCLK = 32MHz
104  */
105 
106 #undef STR71X_PLL1IN_DIV2 /* Don't divide main OSC by two */
107 #define STR71X_PLL1OUT_MUL 16 /* PLL1OUT = 16 * CLK2 */
108 #define STR71X_PLL1OUT_DIV 2 /* PLL1OUT = CLK2 / 2 */
109 #define STR71X_APB1_DIV 1 /* PCLK1 = RCLK */
110 #define STR71X_APB2_DIV 1 /* PCLK2 = RCLK */
111 #define STR71X_MCLK_DIV 1 /* MCLK = RCLK */
