103 /* Digital Frequency Locked Loop configuration
104  * Fdfll = (Fclk * DFLLmul) / DFLLdiv
105  * = 32768 * (48000000/32768) / 1 = 48MHz
106  *
107  * The actual frequency is 47.97MHz due to truncation of the multiplier.
108  * The 48MHz target value is treated as "not-to-exceed" value). Use OSC0
109  * if you need more accuracy (12MHz with a multiplier of 4).
110  *
111  * DFLL0 source options (select one):
112  * BOARD_DFLL0_SOURCE_RCSYS - System RC oscillator
113  * BOARD_DFLL0_SOURCE_OSC32K - 32.768KHz oscillator
114  * BOARD_DFLL0_SOURCE_OSC0 - Oscillator 0
115  * BOARD_DFLL0_SOURCE_RC80M - 80 MHz RC oscillator
116  * BOARD_DFLL0_SOURCE_RC32K - 32 kHz RC oscillator
117  *
118  * NOTE: Nothing must be defined if the DFPLL is not used
119  */
120 
121 #define BOARD_DFLL0_SOURCE_OSC32K 1

123 #define BOARD_DFLL0_MUL (BOARD_DFLL0_TARGET / BOARD_OSC32_FREQUENCY)
124 #define BOARD_DFLL0_DIV 1
125 #define BOARD_DFLL0_FREQUENCY (BOARD_OSC32_FREQUENCY * BOARD_DFLL0_MUL / BOARD_DFLL0_DIV)
