50 /* SD1329 Commands **********************************************************/
51 /* Set column Address.
52  *
53  * This triple byte command specifies column start address and end address of
54  * the display data RAM. This command also sets the column address pointer to
55  * column start address. This pointer is used to define the current read/write
56  * column address in graphic display data RAM. If horizontal address increment
57  * mode is enabled by command 0xa0, after finishing read/write one column data,
58  * it is incremented automatically to the next column address. Whenever the
59  * column address pointer finishes accessing the end column address, it is
60  * reset back to start column address and the row address is incremented to the
61  * next row.
62  *
63  * Byte 1: 0x15
64  * Byte 2: A[5:0]: Start Address, range: 0x00-0x3f
65  * Byte 3: B[5:0]: End Address, range: 0x00-0x3f
66  */
67 
68 #define SSD1329_SET_COLADDR 0x15
69 
70 /* Set Row Address.
71  *
72  * This triple byte command specifies row start address and end address of the
73  * display data RAM. This command also sets the row address pointer to row
74  * start address. This pointer is used to define the current read/write row
75  * address in graphic display data RAM. If vertical address increment mode is
76  * enabled by command 0xa0, after finishing read/write one row data, it is
77  * incremented automatically to the next row address. Whenever the row address
78  * pointer finishes accessing the end row address, it is reset back to start
79  * row address.
80  *
81  * Byte 1: 0x75
82  * Byte 2: A[6:0]: Start Address, range: 0x00-0x7f
83  * Byte 3: B[6:0]: End Address, range: 0x00-0x7f
84  */

235 /* Set Re-map
236  *
237  * This double command has multiple configurations and each bit setting is
238  * described as follows:
239  *
240  * Column Address Remapping (A[0])
241  * This bit is made for increase the flexibility layout of segment signals in
242  * OLED module with segment arranged from left to right (when A[0] is set to 0)
243  * or from right to left (when A[0] is set to 1).
244  *
245  * Nibble Remapping (A[1])
246  * When A[1] is set to 1, the two nibbles of the data bus for RAM access are
247  * re-mapped, such that (D7, D6, D5, D4, D3, D2, D1, D0) acts like (D3, D2, D1,
248  * D0, D7, D6, D5, D4) If this feature works together with Column Address
249  * Re-map, it would produce an effect of flipping the outputs from SEG0-127 to
250  * SEG127-SEG0.
251  *
252  * Address increment mode (A[2])
253  * When A[2] is set to 0, the driver is set as horizontal address incremen
254  * mode. After the display RAM is read/written, the column address pointer is
255  * increased automatically by 1. If the column address pointer reaches column
256  * end address, the column address pointer is reset to column start address and
257  * row address pointer is increased by 1.
258  *
259  * When A[2] is set to 1, the driver is set to vertical address increment mode.
260  * After the display RAM is read/written, the row address pointer is increased
261  * automatically by 1. If the row address pointer reaches the row end address,
262  * the row address pointer is reset to row start address and column address
263  * pointer is increased by 1.
264  *
265  * COM Remapping (A[4])
266  * This bit defines the scanning direction of the common for flexible layout
267  * of common signals in OLED module either from up to down (when A[4] is set to
268  * 0) or from bottom to up (when A[4] is set to 1).
269  *
270  * Splitting of Odd / Even COM Signals (A[6])
271  * This bit is made to match the COM layout connection on the panel. When A[6]
272  * is set to 0, no splitting odd / even of the COM signal is performed. When
273  * A[6] is set to 1, splitting odd / even of the COM signal is performed,
274  * output pin assignment sequence is shown as below (for 128MUX ratio):
275  *
276  * Byte 1: 0xa0
277  * Byte 2: A[7:0]
278  */
279 
280 #define SSD1329_GDDRAM_REMAP 0xa0
281 # define SSD1329_COLADDR_REMAP 0x01 /* A[0]: Enable column re-map */
282 # define SSD1329_NIBBLE_REMAP 0x02 /* A[1]: Enable nibble re-map */
283 # define SSD1329_VADDR_INCR 0x04 /* A[1]: Enable vertical address increment */
284 # define SSD1329_COM_REMAP 0x10 /* A[4]: Enable COM re-map */
285 # define SSD1329_COM_SPLIT 0x40 /* A[6]: Enable COM slip even/odd */

312 /* Set Display Mode - Normal, all on, all off, inverse
313  *
314  * These are single byte commands and are used to set display status to Normal
315  * Display, Entire Display ON, Entire Display OFF or Inverse Display.
316  *
317  * Normal Display (0xa4)

334  */

366 /* Set Phase Length
367  *
368  * In the second byte of this double command, lower nibble and higher nibble is
369  * defined separately. The lower nibble adjusts the phase length of Reset (phase
370  * 1). The higher nibble is used to select the phase length of first pre-charge
371  * phase (phase 2). The phase length is ranged from 1 to 16 DCLK's. RESET for
372  * A[3:0] is set to 3 which means 4 DCLKs selected for Reset phase. POR for
373  * A[7:4] is set to 5 which means 6 DCLKs is selected for first pre-charge
374  * phase. Please refer to Table 9-1 for detail breakdown levels of each step.
375  *
376  * Byte 1: 0xb1
377  * Byte 2: A[3:0]: Phase 1 period of 1~16 DCLKs
378  * A[7:4]: Phase 2 period of 1~16 DCLKs
379  */

424 /* Look Up Table for Gray Scale Pulse width
425  *
426  * This command is used to set each individual gray scale level for the display.
427  * Except gray scale level GS0 that has no pre-charge and current drive, each
428  * gray scale level is programmed in the length of current drive stage pulse
