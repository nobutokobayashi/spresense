511 /*
512  * Optional data for scan request
513  *
514  * Note: these optional parameters are controlling parameters for the
515  * scanning behavior, these do not apply to getting scan results
516  * (SIOCGIWSCAN). Drivers are expected to keep a local BSS table and
517  * provide a merged results with all BSSes even if the previous scan
518  * request limited scanning to a subset, e.g., by specifying an SSID.
519  * Especially, scan results are required to include an entry for the
520  * current BSS if the driver is in Managed mode and associated with an AP.
521  */
522 struct iw_scan_req
523 {
524  uint8_t scan_type; /* IW_SCAN_TYPE_{ACTIVE,PASSIVE} */
525  uint8_t essid_len;

527  * 0 = scan all allowed channels */
528  uint8_t flags; /* reserved as padding; use zero, this may
529  * be used in the future for adding flags
530  * to request different scan behavior */

532  * individual address of a specific BSS */
533

538  * the current association.
