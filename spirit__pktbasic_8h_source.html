40 /* This module can be used to manage the configuration of Spirit Basic packets.
41  * The user can obtain a packet configuration filling the structure struct
42  * spirit_pktbasic_init_s, defining in it some general parameters for the Spirit Basic
43  * packet format. Another structure the user can fill is struct spirit_pktbasic_addr_s
44  * to define the addresses which will be used during the communication. In
45  * addition, functions to set the payload length and the destination address
46  * are provided.
47  *
48  * Example:
49  *
50  * struct spirit_pktbasic_init_s g_pkbasic_init =
51  * {
52  * PKT_PREAMBLE_LENGTH_08BYTES, # preamble length in bytes
53  * PKT_SYNC_LENGTH_4BYTES, # sync word length in bytes
54  * 0x1A2635A8, # sync word
55  * PKT_LENGTH_VAR, # variable or fixed payload length
56  * 7, # length field width in bits (used only for variable length)
57  * PKT_NO_CRC, # CRC mode
58  * PKT_CONTROL_LENGTH_0BYTES, # control field length
59  * S_ENABLE, # address field
60  * S_DISABLE, # FEC
61  * S_ENABLE # whitening
62  * };
63  *
64  * struct spirit_pktbasic_addr_s g_pktbasic_addrinit =
65  * {
66  * S_ENABLE, # enable/disable filtering on my address
67  * 0x34, # my address (address of the current node)
68  * S_DISABLE, # enable/disable filtering on multicast address
69  * 0xee, # multicast address
70  * S_DISABLE, # enable/disable filtering on broadcast address
71  * 0xff # broadcast address
72  * };
73  *
74  * ...
75  *
76  * spirit_pktbasic_initialize(spirit, &g_pkbasic_init);
77  * spirit_pktbasic_addr_initialize(spirit, &g_pktbasic_addrinit);
78  *
79  * ...
80  *
81  * spirit_pktbasic_set_payloadlen(spirit, 20);
82  * spirit_pktcommon_set_nodeaddress(spirit, 0x44);
83  *
84  * ...
85  *
86  * The module provides some other functions that can be used to modify or
87  * read only some configuration parameters.
88  */
89

109 /* SPIRIT Basic Packet Init structure definition. This structure allows
110  * users to set the main options for the Basic packet.
111  */
112 
113 struct spirit_pktbasic_init_s
114 {
115  uint32_t syncwords; /* Specifies the sync words. This parameter is
116  * a uint32_t word with format:
117  * 0x|SYNC1|SYNC2|SYNC3|SYNC4 */
118  uint8_t preamblen; /* Specifies the preamble length. This parameter
119  * can be any value from enum pkt_preamblen_e */
120  uint8_t synclen; /* Specifies the sync word length. The 32bit
121  * word passed (syncwords) will be stored in
122  * the SYNCx registers from the MSB until the
123  * number of bytes in synclen has been stored.
124  * This parameter can be any value of enum
125  * pkt_synlen_e */
126  uint8_t fixvarlen; /* Specifies if a fixed length of packet has to
127  * be used. This parameter can be any value of
128  * enum pkt_fixvar_len_e */
129  uint8_t pktlenwidth; /* Specifies the size of the length of packet
130  * in bits. This field is useful only if the
131  * field fixvarlen is set to BASIC_LENGTH_VAR.
132  * For Basic packets the length width is
133  * log2(max payload length + control length
134  * (0 to 4) + address length (0 or 1)). */
135  uint8_t crcmode; /* Specifies the CRC word length of packet.
136  * This parameter can be any value of enum
137  * pkt_crcmode_e */
138  uint8_t ctrllen; /* Specifies the length of a control field to
139  * be sent. This parameter can be any value
140  * from enum pkt_ctrllen_e */
141  uint8_t txdestaddr; /* Specifies if the destination address has to
142  * be sent. This parameter can be S_ENABLE or
143  * S_DISABLE */
144  uint8_t fec; /* Specifies if FEC has to be enabled. This
145  * parameter can be S_ENABLE or S_DISABLE */
146  uint8_t datawhite; /* Specifies if data whitening has to be
147  * enabled. This parameter can be S_ENABLE or
148  * S_DISABLE */
149 };

156 struct spirit_pktbasic_addr_s
157 {
158  uint8_t destfilter; /* If set RX packet is accepted if its destination
159  * address matches with srcaddr. This parameter
160  * can be S_ENABLE or S_DISABLE */
161  uint8_t srcaddr; /* Specifies the TX packet source address (address
162  * of this node). */
163  uint8_t mcastfilter; /* If set RX packet is accepted if its destination
164  * address matches with mcastaddr. This parameter
165  * can be S_ENABLE or S_DISABLE */
166  uint8_t mcastaddr; /* Specifies the Multicast group address for this
167  * node. */
168  uint8_t bcastfilter; /* If set RX packet is accepted if its destination
169  * address matches with bcastaddr. This parameter
170  * can be S_ENABLE or S_DISABLE */
171  uint8_t bcastaddr; /* Specifies the Broadcast address for this node. */
172 };
