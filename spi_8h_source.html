212  /* These are currently defined feature flags:
213  *
214  * Bit 0: HWFEAT_CRCGENERATION
215  * Hardware CRC generation
216  * Bit 1: HWFEAT_FORCE_CS_INACTIVE_AFTER_TRANSFER
217  * CS rises after every Transmission, also if we provide new data
218  * immediately
219  * Bit 2: HWFEAT_FORCE_CS_ACTIVE_AFTER_TRANSFER
220  * CS does not rise automatically after a transmission, also if
221  * the spi runs out of data (for a long time)
222  * Bit 3: HWFEAT_ESCAPE_LASTXFER
223  * Do not set the LASTXFER-Bit at the last word of the next
224  * exchange, Flag is auto-resetting after the next LASTXFER
225  * condition. (see spi_exchange)
226  * Bit 4: HWFEAT_LSBFIRST
227  * Data transferred LSB first (default is MSB first)
228  */
229 
230 # ifdef CONFIG_SPI_CRCGENERATION
231 # define HWFEAT_CRCGENERATION (1 << 0)
232 # endif
233 
234 # ifdef CONFIG_SPI_CS_CONTROL
235 # define HWFEAT_FORCE_CS_CONTROL_MASK (7 << 1)
236 # define HWFEAT_FORCE_CS_INACTIVE_AFTER_TRANSFER (1 << 1)
237 # define HWFEAT_FORCE_CS_ACTIVE_AFTER_TRANSFER (1 << 2)
238 # define HWFEAT_ESCAPE_LASTXFER (1 << 3)
239 # endif
240 
241 # ifdef CONFIG_SPI_BITORDER
242 # define HWFEAT_MSBFIRST (0 << 4)
243 # define HWFEAT_LSBFIRST (1 << 4)
244 # endif

246 #else

251 # define SPI_HWFEATURES(d,f) (((f) == 0) ? OK : -ENOSYS)

279 /****************************************************************************
280  * Name: SPI_CMDDATA
281  *
282  * Description:
283  * Some devices require and additional out-of-band bit to specify if the
284  * next word sent to the device is a command or data. This is typical, for
285  * example, in "9-bit" displays where the 9th bit is the CMD/DATA bit.
286  * This function provides selection of command or data.
