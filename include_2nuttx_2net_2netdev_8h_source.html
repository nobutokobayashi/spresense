395 /****************************************************************************
396  * Network device driver functions
397  *
398  * These functions are used by a network device driver for interacting
399  * with the NuttX network.
400  *
401  * Process an incoming IP packet.
402  *
403  * This function should be called when the device driver has received
404  * a packet from the network. The packet from the device driver must
405  * be present in the d_buf buffer, and the length of the packet
406  * should be placed in the d_len field.
407  *
408  * When the function returns, there may be an outbound packet placed
409  * in the d_buf packet buffer. If so, the d_len field is set to
410  * the length of the packet. If no packet is to be sent out, the
411  * d_len field is set to 0.
412  *
413  * The usual way of calling the function is presented by the source
414  * code below.
415  *
416  * dev->d_len = devicedriver_poll();
417  * if (dev->d_len > 0)
418  * {
419  * ipv4_input(dev);
420  * if (dev->d_len > 0)
421  * {
422  * devicedriver_send();
423  * }
424  * }
425  *
426  * Note: If you are writing a network device driver that needs ARP
427  * (Address Resolution Protocol), e.g., when running the network over
428  * Ethernet, you will need to call the network ARP code before calling
429  * this function:
430  *
431  * #define BUF ((struct eth_hdr_s *)&dev->d_buf[0])
432  * dev->d_len = ethernet_devicedrver_poll();
433  * if (dev->d_len > 0)
434  * {
435  * if (BUF->type == HTONS(ETHTYPE_IP))
436  * {
437  * arp_ipin();
438  * ipv4_input(dev);
439  * if (dev->d_len > 0)
440  * {
441  * arp_out();
442  * devicedriver_send();
443  * }
444  * }
445  * else if (BUF->type == HTONS(ETHTYPE_ARP))
446  * {
447  * arp_arpin();
448  * if (dev->d_len > 0)
449  * {
450  * devicedriver_send();
451  * }
452  * }
453  *
454  ****************************************************************************/
455

458 #endif

472 /****************************************************************************
473  * Polling of connections
474  *
475  * These functions will traverse each active network connection structure
476  * and perform appropriate operations: devif_timer() will perform TCP timer
477  * operations (and UDP polling operations); devif_poll() will perform TCP
478  * and UDP polling operations. The CAN driver MUST implement logic to
479  * periodically call devif_timer(); devif_poll() may be called asynchronously
480  * from the network driver can accept another outgoing packet.
481  *
482  * In both cases, these functions will call the provided callback function
483  * for every active connection. Polling will continue until all connections
484  * have been polled or until the user-supplied function returns a non-zero
485  * value (which it should do only if it cannot accept further write data).
486  *
487  * When the callback function is called, there may be an outbound packet
488  * waiting for service in the device packet buffer, and if so the d_len field
489  * is set to a value larger than zero. The device driver should then send
490  * out the packet.
491  *
492  * Example:
493  * int driver_callback(FAR struct net_driver_s *dev)
494  * {
495  * if (dev->d_len > 0)
496  * {
497  * devicedriver_send();
498  * return 1; <-- Terminates polling if necessary
499  * }
500  * return 0;
501  * }
502  *
503  * ...
504  * devif_poll(dev, driver_callback);
505  *
506  * Note: If you are writing a network device driver that needs ARP (Address
507  * Resolution Protocol), e.g., when running the networ over Ethernet, you
508  * will need to call the arp_out() function in the callback function
509  * before sending the packet:
510  *
511  * int driver_callback(FAR struct net_driver_s *dev)
512  * {
513  * if (dev->d_len > 0)
514  * {
515  * arp_out();
516  * devicedriver_send();
517  * return 1; <-- Terminates polling if necessary
518  * }
519  *
520  * return 0;
521  * }
522  *
523  ****************************************************************************/
524 
525 int devif_poll(FAR struct net_driver_s *dev, devif_poll_callback_t callback);
526 int devif_timer(FAR struct net_driver_s *dev, devif_poll_callback_t callback);

528 /****************************************************************************
529  * Name: neighbor_out
530  *
531  * Description:
532  * This function should be called before sending out an IPv6 packet. The
533  * function checks the destination IPv6 address of the IPv6 packet to see
534  * what Ethernet MAC address that should be used as a destination MAC
535  * address on the Ethernet.
536  *
537  * If the destination IPv6 address is in the local network (determined
538  * by logical ANDing of netmask and our IPv6 address), the function
539  * checks the Neighbor Table to see if an entry for the destination IPv6
540  * address is found. If so, an Ethernet header is pre-pended at the
541  * beginning of the packet and the function returns.
542  *
543  * If no Neighbor Table entry is found for the destination IPv6 address,
544  * the packet in the d_buf[] is replaced by an ICMPv6 Neighbor Solict
545  * request packet for the IPv6 address. The IPv6 packet is dropped and
546  * it is assumed that the higher level protocols (e.g., TCP) eventually
547  * will retransmit the dropped packet.
548  *
549  * Upon return in either the case, a packet to be sent is present in the
550  * d_buf[] buffer and the d_len field holds the length of the Ethernet
551  * frame that should be transmitted.
552  *
553  ****************************************************************************/
554 
555 #ifdef CONFIG_NET_IPv6
556 void neighbor_out(FAR struct net_driver_s *dev);
557 #endif /* CONFIG_NET_IPv6 */
