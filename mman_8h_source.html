94 /* If the Advisory Information and either the Memory Mapped Files or Shared
95  * Memory Objects options are supported, values for advice used by
96  * posix_madvise() are defined as follows:
97  *
98  * POSIX_MADV_NORMAL
99  * The application has no advice to give on its behavior with respect to
100  * the specified range. It is the default characteristic if no advice is
101  * given for a range of memory.
102  * POSIX_MADV_SEQUENTIAL
103  * The application expects to access the specified range sequentially from
104  * lower addresses to higher addresses.
105  * POSIX_MADV_RANDOM
106  * The application expects to access the specified range in a random order.
107  * POSIX_MADV_WILLNEED
108  * The application expects to access the specified range in the near
109  * future.
110  * POSIX_MADV_DONTNEED
111  * The application expects that it will not access the specified range in
112  * the near future.
113  */
114 
115 #define POSIX_MADV_NORMAL (0)
116 #define POSIX_MADV_SEQUENTIAL (1)
117 #define POSIX_MADV_RANDOM (2)
118 #define POSIX_MADV_WILLNEED (3)
