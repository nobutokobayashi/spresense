292 #define DMA_STREAM_DMEIF_BIT (1 << 2) /* Bit 2: Stream direct mode error interrupt flag */
293 #define DMA_STREAM_TEIF_BIT (1 << 3) /* Bit 3: Stream Transfer Error flag */

396 /* DMA Stream mapping. Each DMA stream has a mapping to several possible
397  * sources/sinks of data. The requests from peripherals assigned to a stream
398  * are simply OR'ed together before entering the DMA block. This means that only
399  * one request on a given stream can be enabled at once.
400  *
401  * Alternative stream selections are provided with a numeric suffix like _1, _2, etc.
402  * The DMA driver, however, will use the pin selection without the numeric suffix.
403  * Additional definitions are required in the board.h file. For example, if
404  * SPI3_RX connects via DMA STREAM0, then following should be application-specific
405  * mapping should be used:
406  *
407  * #define DMAMAP_SPI3_RX DMAMAP_SPI3_RX_1
408  */
409 
410 #define STM32_DMA_MAP(d,s,c) ((d) << 7 | (s) << 4 | (c))
411 #define STM32_DMA_CONTROLLER(m) (((m) >> 7) & 1)
412 #define STM32_DMA_STREAM(m) (((m) >> 4) & 7)
413 #define STM32_DMA_CHANNEL(m) ((m) & 0xf)
