61  * Other settings that effect the driver: CONFIG_DISABLE_POLL

63  * CONFIG_INPUT_STMPE811

68  * Enables support for the I2C interface

70  * Can be defined to support multiple STMPE811 devices on board.
71  * CONFIG_STMPE811_NPOLLWAITERS
72  * Maximum number of threads that can be waiting on poll() (ignored if
73  * CONFIG_DISABLE_POLL is set).
74  * CONFIG_STMPE811_TSC_DISABLE
75  * Disable driver touchscreen functionality.
76  * CONFIG_STMPE811_ADC_DISABLE
77  * Disable driver ADC functionality.
78  * CONFIG_STMPE811_GPIO_DISABLE
79  * Disable driver GPIO functionality.
80  * CONFIG_STMPE811_GPIOINT_DISABLE
81  * Disable driver GPIO interrupt functionality (ignored if GPIO functionality is
82  * disabled).
83  * CONFIG_STMPE811_SWAPXY
84  * Reverse the meaning of X and Y to handle different LCD orientations.
85  * CONFIG_STMPE811_TEMP_DISABLE
86  * Disable driver temperature sensor functionality.
87  * CONFIG_STMPE811_REGDEBUG
88  * Enable very low register-level debug output. Requires CONFIG_DEBUG_FEATURES.
89  * CONFIG_STMPE811_THRESHX and CONFIG_STMPE811_THRESHY
90  * STMPE811 touchscreen data comes in a a very high rate. New touch positions
91  * will only be reported when the X or Y data changes by these thresholds.
92  * This trades reduces data rate for some loss in dragging accuracy. The
93  * STMPE811 is configure for 12-bit values so the raw ranges are 0-4095. So
94  * for example, if your display is 320x240, then THRESHX=13 and THRESHY=17
95  * would correspond to one pixel. Default: 12
96  */

401 /* GPIO Configuration ***********************************************************************/
402 /* The STMPE811 GPIO interfaces take an 8-bit bit-encoded parameter to describe the GPIO pin.
403  * The following definitions describe the bit-encoding of that parameter.
404  *
405  * 7654 3210
406  * ---- ----
407  * DIRF VPPP
408  *
409  * Input Pins: 1IRF .PPP
410  *
411  * Output Pins: 0... VPPP
412  *
413  * Bits 7 is the pin direction.
414  */
415 
416 #define STMPE811_GPIO_DIR (1 << 7) /* Bit7: Direction bit */
417 #define STMPE811_GPIO_INPUT (1 << 7) /* Input pin (possibly interrupting) */
418 #define STMPE811_GPIO_OUTPUT (0) /* Configure as in output */
419 
420 /* Bit 6 indicates that the pin will generate an interrupt (inputs only) */
421 
422 #define STMPE811_GPIO_IN (1 << 6) /* Bit 6: Input interrupting pin */
423 
424 /* The bits 4-5 select the rising and/or the falling edge detection. */
425 
426 #define STMPE811_GPIO_RISING (1 << 5) /* Bit 5: Input interrupting pin */
427 #define STMPE811_GPIO_FALLING (1 << 4) /* Bit 4: Input interrupting pin */
428 
429 /* Bit 3 is the initial value for output pins */
430 
431 #define STMPE811_GPIO_VALUE (1 << 3) /* Bit 3: The initial value of an output pin */
432 # define STMPE811_GPIO_ONE (1 << 3) /* Bit 3: The initial value is logic 1 */
433 # define STMPE811_GPIO_ZERO (0) /* Bit 3: The initial value is logic 0 */
434 
435 /* Bits 0-2 is the pin number */
436
