237 /* Current state of the worker thread */
238 
239 #define USBMSC_STATE_NOTSTARTED (0) /* Thread has not yet been started */
240 #define USBMSC_STATE_STARTED (1) /* Started, but is not yet initialized */
241 #define USBMSC_STATE_IDLE (2) /* Started and waiting for commands */
242 #define USBMSC_STATE_CMDPARSE (3) /* Processing a received command */
243 #define USBMSC_STATE_CMDREAD (4) /* Processing a SCSI read command */
244 #define USBMSC_STATE_CMDWRITE (5) /* Processing a SCSI write command */
245 #define USBMSC_STATE_CMDFINISH (6) /* Finish command processing */
246 #define USBMSC_STATE_CMDSTATUS (7) /* Processing the final status of the command */
247 #define USBMSC_STATE_TERMINATED (8) /* Thread has exitted */
248 
249 /* Event communicated to worker thread */
250 
251 #define USBMSC_EVENT_NOEVENTS (0) /* There are no outstanding events */
252 #define USBMSC_EVENT_READY (1 << 0) /* Initialization is complete */
253 #define USBMSC_EVENT_RDCOMPLETE (1 << 1) /* A read has completed there is data to be processed */
254 #define USBMSC_EVENT_WRCOMPLETE (1 << 2) /* A write has completed and a request is available */
255 #define USBMSC_EVENT_TERMINATEREQUEST (1 << 3) /* Shutdown requested */
256 #define USBMSC_EVENT_DISCONNECT (1 << 4) /* USB disconnect received */
257 #define USBMSC_EVENT_RESET (1 << 5) /* USB storage setup reset received */
258 #define USBMSC_EVENT_CFGCHANGE (1 << 6) /* USB setup configuration change received */
259 #define USBMSC_EVENT_IFCHANGE (1 << 7) /* USB setup interface change received */
260 #define USBMSC_EVENT_ABORTBULKOUT (1 << 8) /* SCSI receive failure */

422 
423  /* SCSI command processing */
424 
425  struct usbmsc_lun_s *lun; /* Currently selected LUN */
426  struct usbmsc_lun_s *luntab; /* Allocated table of all LUNs */
427  uint8_t cdb[USBMSC_MAXCDBLEN]; /* Command data (cdb[]) from CBW */
428  uint8_t phaseerror:1; /* Need to send phase sensing status */
