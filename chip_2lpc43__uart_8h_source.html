215 /* IER (DLAB =0) Interrupt Enable Register */
216 
217 #define UART_IER_RBRIE (1 << 0) /* Bit 0: RBR Interrupt Enable */
218 #define UART_IER_THREIE (1 << 1) /* Bit 1: THRE Interrupt Enable */

229 /* IIR Interrupt ID Register */
230 
231 #define UART_IIR_INTSTATUS (1 << 0) /* Bit 0: Interrupt status (active low) */
232 #define UART_IIR_INTID_SHIFT (1) /* Bits 1-3: Interrupt identification */
233 #define UART_IIR_INTID_MASK (7 << UART_IIR_INTID_SHIFT)
234 # define UART_IIR_INTID_MSI (0 << UART_IIR_INTID_SHIFT) /* Modem Status (UART only) */
235 # define UART_IIR_INTID_THRE (1 << UART_IIR_INTID_SHIFT) /* THRE Interrupt */
236 # define UART_IIR_INTID_RDA (2 << UART_IIR_INTID_SHIFT) /* 2a - Receive Data Available (RDA) */
237 # define UART_IIR_INTID_RLS (3 << UART_IIR_INTID_SHIFT) /* 1 - Receive Line Status (RLS) */
238 # define UART_IIR_INTID_CTI (6 << UART_IIR_INTID_SHIFT) /* 2b - Character Time-out Indicator (CTI) */
239  /* Bits 4-5: Reserved */
240 #define UART_IIR_FIFOEN_SHIFT (6) /* Bits 6-7: Copies of FCR[0] */
241 #define UART_IIR_FIFOEN_MASK (3 << UART_IIR_FIFOEN_SHIFT)
242 #define UART_IIR_ABEOINT (1 << 8) /* Bit 8: End of auto-baud interrupt */
243 #define UART_IIR_ABTOINT (1 << 9) /* Bit 9: Auto-baud time-out interrupt */
244  /* Bits 10-31: Reserved */
245 /* FCR FIFO Control Register */

255 # define UART_FCR_RXTRIGGER_4 (1 << UART_FCR_RXTRIGGER_SHIFT) /* Trigger level 1 (4 chars) */

271 # define UART_LCR_PS_ODD (0 << UART_LCR_PS_SHIFT) /* Odd parity */
272 # define UART_LCR_PS_EVEN (1 << UART_LCR_PS_SHIFT) /* Even Parity */

274 # define UART_LCR_PS_STICKY0 (3 << UART_LCR_PS_SHIFT) /* Forced "0" stick parity */
275 #define UART_LCR_BRK (1 << 6) /* Bit 6: Break Control */
