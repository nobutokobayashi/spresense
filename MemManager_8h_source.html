73 static inline void* translateVaToPa(void* addr)
74 {
75  uint32_t tileId;
76  uint32_t tileVal;
77  uint32_t cpuId;
78  uint32_t reg;
79  uint32_t pa;
80  uint32_t va;
81 
82  va = (uint32_t)addr;
83  tileId = (va >> 16) & 0xf;
84  cpuId = *(volatile uint32_t *)((0x4c000000 | 0x02002000) + 0x40);
85  reg = (0x02012000 + 0x04) + (0x04 * (tileId / 2)) + ((cpuId - 2) * 0x20);
86  pa = *(volatile uint32_t *)(reg);
87  tileVal = ((pa >> ((tileId & 0x1) * 16)) & 0x01ff) << 16;
88 
89  return (void *)(0x0c000000 | tileVal | (va & 0xffff));
90 }
