138 /* HSMCI clocking
139  *
140  * Multimedia Card Interface clock (MCCK or MCI_CK) is Master Clock (MCK)
141  * divided by (2*(CLKDIV+1)).
142  *
143  * MCI_SPEED = MCK / (2*(CLKDIV+1))
144  * CLKDIV = MCK / MCI_SPEED / 2 - 1
145  *
146  * Where CLKDIV has a range of 0-255.
147  */
148 
149 #ifdef CONFIG_SAM4EEK_120MHZ
150  /* MCK = 120MHz, CLKDIV = 149 w/o CLKODD, MCI_SPEED = 120MHz / (2*149 + 0 + 2) = 400 KHz */
151 
152 # define HSMCI_INIT_CLKDIV (149 << HSMCI_MR_CLKDIV_SHIFT)
153 
154  /* MCK = 120MHz, CLKDIV = 2 w/o CLKODD, MCI_SPEED = 120MHz / (2*2 + 0 + 2) = 20 MHz */
155 
156 # define HSMCI_MMCXFR_CLKDIV (3 << HSMCI_MR_CLKDIV_SHIFT)

207 /* LED definitions ******************************************************************/
208 /* The SAM4E-EK board has three, user-controllable LEDs labelled D2 (blue),
209  * D3 (amber), and D4 (green) on the board. Usage of these LEDs is defined
210  * in include/board.h and src/up_leds.c. They are encoded as follows:
211  *
212  * SYMBOL Meaning D3* D2 D4
213  * ------------------- ----------------------- ------- ------- -------
214  * LED_STARTED NuttX has been started OFF OFF OFF
215  * LED_HEAPALLOCATE Heap has been allocated OFF OFF ON
216  * LED_IRQSENABLED Interrupts enabled OFF ON OFF
217  * LED_STACKCREATED Idle stack created OFF ON ON
218  * LED_INIRQ In an interrupt** N/C FLASH N/C
219  * LED_SIGNAL In a signal handler*** N/C N/C FLASH
220  * LED_ASSERTION An assertion failed FLASH N/C N/C
221  * LED_PANIC The system has crashed FLASH N/C N/C
222  *
223  * * If D2 and D4 are statically on, then NuttX probably failed to boot
224  * and these LEDs will give you some indication of where the failure was
225  * ** The normal state is D3=OFF, D4=ON and D2 faintly glowing. This faint
226  * glow is because of timer interrupts that result in the LED being
227  * illuminated on a small proportion of the time.
228  * *** D4 may also flicker normally if signals are processed.
229  */
230 
231 #define LED_STARTED 0 /* LED0=OFF LED1=OFF LED2=OFF */
232 #define LED_HEAPALLOCATE 1 /* LED0=OFF LED1=OFF LED2=ON */
233 #define LED_IRQSENABLED 2 /* LED0=OFF LED1=ON LED2=OFF */
234 #define LED_STACKCREATED 3 /* LED0=OFF LED1=ON LED2=ON */
235 
236 #define LED_INIRQ 4 /* LED0=XXX LED1=TOG LED2=XXX */
237 #define LED_SIGNAL 5 /* LED0=XXX LED1=XXX LED2=TOG */
238 #define LED_ASSERTION 6 /* LED0=TOG LED1=XXX LED2=XXX */
239 #define LED_PANIC 7 /* LED0=TOG LED1=XXX LED2=XXX */
