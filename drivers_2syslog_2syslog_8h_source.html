76 /****************************************************************************
77  * Name: syslog_dev_initialize
78  *
79  * Description:
80  * Initialize to use the character device (or file) at
81  * CONFIG_SYSLOG_DEVPATH as the SYSLOG sink.
82  *
83  * One power up, the SYSLOG facility is non-existent or limited to very
84  * low-level output. This function may be called later in the
85  * intialization sequence after full driver support has been initialized.
86  * (via syslog_initialize()) It installs the configured SYSLOG drivers
87  * and enables full SYSLOGing capability.
88  *
89  * NOTE that this implementation excludes using a network connection as
90  * SYSLOG device. That would be a good extension.
91  *
92  * Input Parameters:
93  * devpath - The full path to the character device to be used.
94  * oflags - File open flags
95  * mode - File open mode (only if oflags include O_CREAT)
96  *
97  * Returned Value:
98  * Zero (OK) is returned on success; a negated errno value is returned on
99  * any failure.
100  *
101  ****************************************************************************/
102 
103 #if CONFIG_NFILE_DESCRIPTORS > 0
104 int syslog_dev_initialize(FAR const char *devpath, int oflags, int mode);
105 #endif

159 /****************************************************************************
160  * Name: syslog_console_channel
161  *
162  * Description:
163  * Configure to use the character device (or file) at /dev/console as the
164  * SYSLOG channel.
165  *
166  * This tiny function is simply a wrapper around syslog_dev_initialize()
167  * and syslog_channel(). It calls syslog_dev_initialize() to configure
168  * the character device at /dev/console then calls syslog_channel() to
169  * use that device as the SYSLOG output channel.

171  * NOTE interrupt level SYSLOG output will be lost in the general case
172  * unless the interrupt buffer is used. As a special case: If the serial
173  * console is used and the architecture provides up_putc(), the interrupt
174  * level output will be directed to up_putc() is the interrupt buffer is
175  * disabled.
176  *
177  * Input Parameters:
178  * None
179  *
180  * Returned Value:
181  * Zero (OK) is returned on success; a negated errno value is returned on
182  * any failure.
183  *
184  ****************************************************************************/

190 /****************************************************************************
191  * Name: syslog_add_intbuffer
192  *
193  * Description:
194  * Add one more character to the interrupt buffer. In the event of
195  * buffer overlowed, the character will be dropped. The indication
196  * "[truncated]\n" will be appended to the end of the interrupt buffer.
197  *
198  * Input Parameters:
199  * ch - The character to add to the interrupt buffer (must be positive).
200  *
201  * Returned Value:
202  * Zero success, the character is echoed back to the caller. A negated
203  * errno value is returned on any failure.
204  *
205  * Assumptions:
206  * Called only from interrupt handling logic; Interrupts will be disabled.
207  *
208  ****************************************************************************/
