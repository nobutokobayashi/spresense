550 static inline int mac802154_takesem(sem_t *sem, bool allowinterrupt)
551 {
552  int ret;
553  do
554  {
555  /* Take a count from the semaphore, possibly waiting */
556 
557  ret = sem_wait(sem);
558  if (ret < 0)
559  {
560  /* EINTR is the only error that we expect */
561 
562  DEBUGASSERT(get_errno() == EINTR);
563 
564  if (allowinterrupt)
565  {
566  return -EINTR;
567  }
568  }
569  }
570  while (ret != OK);
571 
572  return OK;
573 }

629  uint32_t symbols)
630 {
631  union ieee802154_attr_u attrval;
632  uint32_t ret;
633 
634  /* First, get the symbol duration from the radio layer. Symbol duration is
635  * returned in picoseconds to ensure precision is kept when multiplying to
636  * get overall times.
637  */
638 
639  priv->radio->getattr(priv->radio, IEEE802154_ATTR_PHY_SYMBOL_DURATION,
640  &attrval);
641 
642  /* After this step, ret represents microseconds */
643 
644  ret = ((uint64_t)attrval.phy.symdur_picosec * symbols) / (1000 * 1000);
645 
646  /* This method should only be used for things that can be late. For instance,
647  * it's always okay to wait a little longer before disabling your receiver.
648  * Therefore, we force the tick count to round up.
649  */
650 
651  if (ret % USEC_PER_TICK == 0)
652  {
653  ret = ret/USEC_PER_TICK;
654  }
655  else
656  {
657  ret = ret/USEC_PER_TICK;
658  ret++;
659  }
660 
661  return ret;
662 }
