51 /* ENCX24J600 Commands ********************************************************/
52 
53 /* The SPI opcodes are divided into four families:
54  *
55  * Single Byte: Direct opcode instructions; designed for task-oriented SFR
56  * operations with no data returned
57  *
58  * Two-Byte: Direct opcode instruction; designed for SFR operation with byte
59  * data returned
60  *
61  * Three-Byte: Opcode with word length argument; includes read and write
62  * operations, designed for pointer manipulation with word length data returned
63  *
64  * N-Byte: Opcode with one or more bytes of argument; includes read and write
65  * operations designed for general memory space access with one or more bytes of
66  * data returned
67  */
68
