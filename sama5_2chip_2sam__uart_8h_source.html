282 /* UART Control Register */
283 
284 #define UART_CR_RSTRX (1 << 2) /* Bit 2: Reset Receiver (Common) */
285 #define UART_CR_RSTTX (1 << 3) /* Bit 3: Reset Transmitter (Common) */
286 #define UART_CR_RXEN (1 << 4) /* Bit 4: Receiver Enable (Common) */
287 #define UART_CR_RXDIS (1 << 5) /* Bit 5: Receiver Disable (Common) */
288 #define UART_CR_TXEN (1 << 6) /* Bit 6: Transmitter Enable (Common) */
289 #define UART_CR_TXDIS (1 << 7) /* Bit 7: Transmitter Disable (Common) */
290 #define UART_CR_RSTSTA (1 << 8) /* Bit 8: Reset Status Bits (Common) */
291 
292 #if defined(ATSAMA5D2)
293 # define UART_CR_STTTO (1 << 11) /* Bit 11: Start Time-out (USART UART mode only) */
294 # define UART_REQCLR_STTBRK (1 << 12) /* Bit 12: Request Clear */
295 #elif defined(ATSAMA5D3) ||defined(ATSAMA5D4)
296 # define UART_CR_STTBRK (1 << 9) /* Bit 9: Start Break (USART UART mode only) */
297 # define UART_CR_STPBRK (1 << 10) /* Bit 10: Stop Break (USART UART mode only) */
298 # define UART_CR_STTTO (1 << 11) /* Bit 11: Start Time-out (USART UART mode only) */
299 # define UART_CR_SENDA (1 << 12) /* Bit 12: Send Address (USART UART mode only) */
300 # define UART_CR_RSTIT (1 << 13) /* Bit 13: Reset Iterations (USART UART mode only) */
301 # define UART_CR_RSTNACK (1 << 14) /* Bit 14: Reset Non Acknowledge (USART UART mode only) */
302 # define UART_CR_RETTO (1 << 15) /* Bit 15: Rearm Time-out (USART UART mode only) */
303 # define UART_CR_RTSEN (1 << 18) /* Bit 18: Request to Send Enable (USART UART mode only) */
304 # define UART_CR_FCS (1 << 18) /* Bit 18: Force SPI Chip Select (USART SPI mode only) */
305 # define UART_CR_RTSDIS (1 << 19) /* Bit 19: Request to Send Disable (USART UART mode only) */
306 # define UART_CR_RCS (1 << 19) /* Bit 19: Release SPI Chip Select (USART SPI mode only) */
307 #endif

309 /* UART Mode Register and USART Mode Register (UART MODE)
310  *
311  * NOTES:
312  * (1) Common to UART and USART (all modes)
313  * (2) Common to UART and USART (UART mode only)
314  * (3) USART only (all modes)
315  * (4) USART only (UART mode only)
316  * (5) USART only (SPI mode only)
317  */
318 
319 #if defined(ATSAMA5D2)
320 # define UART_MR_FILTER (1 << 4) /* Bit 4: Receiver Digital Filter */
321 #elif defined(ATSAMA5D3) ||defined(ATSAMA5D4)
322 # define UART_MR_MODE_SHIFT (0) /* Bits 0-3: (3) */
323 # define UART_MR_MODE_MASK (15 << UART_MR_MODE_SHIFT)
324 # define UART_MR_MODE_NORMAL (0 << UART_MR_MODE_SHIFT) /* Normal */
325 # define UART_MR_MODE_RS485 (1 << UART_MR_MODE_SHIFT) /* RS485 */
326 # define UART_MR_MODE_HWHS (2 << UART_MR_MODE_SHIFT) /* Hardware Handshaking */
327 # define UART_MR_MODE_ISO7816_0 (4 << UART_MR_MODE_SHIFT) /* IS07816 Protocol: T = 0 */
