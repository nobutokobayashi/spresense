105 /* HSMCI clocking
106  *
107  * Multimedia Card Interface clock (MCCK or MCI_CK) is Master Clock (MCK)
108  * divided by (2*(CLKDIV+1)).
109  *
110  * MCI_SPEED = MCCK / (2*(CLKDIV+1))
111  * CLKDIV = MCCK / MCI_SPEED / 2 - 1
112  *
113  * Where CLKDIV has a range of 0-255.
114  */
115 
116 /* MCK = 84MHz, CLKDIV = 104, MCI_SPEED = 84MHz / 2 * (104+1) = 400 KHz */

118 #define HSMCI_INIT_CLKDIV (104 << HSMCI_MR_CLKDIV_SHIFT)

120 /* MCK = 84MHz, CLKDIV = 2, MCI_SPEED = 84MHz / 2 * (2+1) = 14 MHz */
