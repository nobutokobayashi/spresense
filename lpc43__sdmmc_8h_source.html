132 #define SDMMC_CTRL_CNTLRRESET (1 << 0) /* Bit 0: Reset Module controller */
133 #define SDMMC_CTRL_FIFORESET (1 << 1) /* Bit 1: Reset to data FIFO To reset FIFO pointers */
134 #define SDMMC_CTRL_DMARESET (1 << 2) /* Bit 2: Reset internal DMA interface control logic */

136 #define SDMMC_CTRL_INTENABLE (1 << 4) /* Bit 4: Enable interrupts */
137  /* Bit 5: Reserved */

139 #define SDMMC_CTRL_SENDIRQRESP (1 << 7) /* Bit 7: Send auto IRQ response */
140 #define SDMMC_CTRL_ABORTREAD (1 << 8) /* Bit 8: Reset data state-machine (suspend sequence) */
141 #define SDMMC_CTRL_SENDCCSD (1 << 9) /* Bit 9: Send CCSD to CE-ATA device */
142 #define SDMMC_CTRL_AUTOSTOP (1 << 10) /* Bit 10: Send STOP after CCSD to CE-ATA device */
143 #define SDMMC_CTRL_CEATAINT (1 << 11) /* Bit 11: CE-ATA device interrupts enabled */
144  /* Bits 12-15: Reserved */
145 #define SDMMC_CTRL_CDVA0 (1 << 16) /* Bit 16: Controls SD_VOLT0 pin */
146 #define SDMMC_CTRL_CDVA1 (1 << 17) /* Bit 17: Controls SD_VOLT1 pin */
147 #define SDMMC_CTRL_CDVA2 (1 << 18) /* Bit 18: Controls SD_VOLT2 pin */
148  /* Bits 19-23: Reserved */
149  /* Bit 24: Reserved - always write it as 0 */
150 #define SDMMC_CTRL_INTDMA (1 << 25) /* Bit 25: SD/MMC DMA use */
151  /* Bits 26-31: Reserved */
