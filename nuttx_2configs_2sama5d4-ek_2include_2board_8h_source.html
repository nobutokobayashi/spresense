102 /* LED definitions ******************************************************************/
103 /* There are 3 LEDs on the SAMA5D4-EK:
104  *
105  * ------------------------------ ------------------- -------------------------
106  * SAMA5D4 PIO SIGNAL USAGE
107  * ------------------------------ ------------------- -------------------------
108  * PE28/NWAIT/RTS4/A19 1Wire_PE28 1-WIRE ROM, LCD, D8 (green)
109  * PE8/A8/TCLK3/PWML3 LED_USER_PE8 LED_USER (D10)
110  * PE9/A9/TIOA2 LED_POWER_PE9 LED_POWER (D9, Red)
111  * ------------------------------ ------------------- -------------------------
112  *
113  * - D8: D8 is shared with other functions and cannot be used if the 1-Wire ROM
114  * is used. I am not sure of the LCD function, but the LED may not be available
115  * if the LCD is used either. We will avoid using D8 just for simplicity.
116  * - D10: Nothing special here. A low output illuminates.
117  * - D9: The Power ON LED. Connects to the via an IRLML2502 MOSFET. This LED will
118  * be on when power is applied but otherwise a low output value will turn it
119  * off.
120  */
121

133 /* These LEDs are not used by the board port unless CONFIG_ARCH_LEDS is
134  * defined. In that case, the usage by the board port is defined in
135  * include/board.h and src/sam_leds.c. The LEDs are used to encode OS-related
136  * events as follows:
137  *
138  * SYMBOL Val Meaning LED state
139  * Blue Red
140  * ----------------- --- ----------------------- -------- -------- */
141 #define LED_STARTED 0 /* NuttX has been started OFF OFF */
142 #define LED_HEAPALLOCATE 0 /* Heap has been allocated OFF OFF */
143 #define LED_IRQSENABLED 0 /* Interrupts enabled OFF OFF */
144 #define LED_STACKCREATED 1 /* Idle stack created ON OFF */
145 #define LED_INIRQ 2 /* In an interrupt No change */
146 #define LED_SIGNAL 2 /* In a signal handler No change */
147 #define LED_ASSERTION 2 /* An assertion failed No change */
148 #define LED_PANIC 3 /* The system has crashed OFF Blinking */
149 #undef LED_IDLE /* MCU is is sleep mode Not used */

236  *

250 /* PCK0 and PCK1 are not currently used, but the PCK logic wants these definitions
251  * anyway. The assignments here are arbitrary and will not be used (at least not
252  * until we implement ISI of HDMI).
253  *
254  * PIO_PMC_PCK0_1: PB26 is used by I2S with the WM8904 (AUDIO_RK0_PB26)
255  * PIO_PMC_PCK0_2: PD8 is the HDMI MCLK (HDMI_MCK_PD8)
256  * PIO_PMC_PCK0_3: PA24 is used for the LCD backlight (LCD_PWM_PA24)
257  *
258  * PIO_PMC_PCK1_1: PD31 goes to the expansion interface and is not used on-board
259  * (EXP_PD31).
260  * PIO_PMC_PCK1_2: PC24 is used for ISI data (ISI_D5)
261  * PIO_PMC_PCK1_3: PC4 is ISI_MCK_PC4, MCI0_CK_PC4, EXP_PC4
262  */
