54 /* Clocking *************************************************************************/
55 /* The LaunchXL-TMS57004 has a 16 MHz external crystal. */
56 
57 #define BOARD_FCLKIN_FREQUENCY 16000000 /* 16 MHz crystal frequency */
58 
59 /* The maximum frequency for the TMS570LS0432PZ is 80 MHz.
60  *
61  * REFCLKDIV controls input clock divider:
62  *
63  * NR = REFCLKDIV+1
64  * Fintclk = Fclkin / NR
65  *
66  * PLLMUL controls multipler on divided input clock (Fintclk):
67  *
68  * Non-modulated:
69  * NF = (PLLMUL + 256) / 256
70  * Modulated:
71  * NF = (PLLMUL + MULMOD + 256) / 256
72  *
73  * Foutputclk = Fintclk x NF (150MHz - 550MHz)
74  *
75  * ODPLL controls internal PLL output divider:
76  *
77  * OD = ODPLL+1
78  * Fpostodclk = Foutputclock / OD
79  *
80  * Final divisor, R, controls PLL output:
81  *
82  * R = PLLDIV + 1
83  * Fpllclock = Fpostodclk / R
84  *
85  * Or:
86  *
87  * Fpllclock = = (Fclkin / NR) x NF / OD / R
88  *
89  * In this case, we have:
90  *
91  * Fclkin = 16,000,000
92  * NR = 6 (REFCLKDIV=5)
93  * NF = 120 (PLLMUL = 119 * 256)
94  * OD = 1 (ODPLL = 0)
95  * R = 2 (PLLDIV=1)
96  *
97  * Then:
98  *
99  * Fintclk = 16 MHz / 6 = 2.667 MHz
100  * Foutputclock = 2.667 MHz * 120 = 320 MHz
101  * Fpostodclock = 320 MHz / 2 = 160 MHz
102  * Fpllclock = 160 MHz / 2 = 80 MHz
103  */
104

181 /* LED definitions ******************************************************************/
182 /* LEDs
183  *
184  * The launchpad has several LEDs:
185  *
186  * - LEd D1 (white) that connects to the USB +5V supply,
187  * - LED D10 (red) that connects to the TMS570's NERROR pin,
188  * - D5 (blue), D6 (blue), and D8 (blue) connect to the XDS100 FT2322,
189  * - D7 (blue) connects to the XSD100 CPLD, and
190  * - Two white, user LEDs labeled D12 that connects to the NHET08
191  * pin and D11 that connects to GIOA2.
192  *
193  * NHET08 is one of 32 N2HET pins than can be available to the user if not used by
194  * N2HET. This implementation, however, uses only the single LED driven by GIOA2.
195  * That LED is tied to ground and illuminated with a high level output value.
196  */
