76 /* Memory barriers for use with NuttX spinlock logic
77  *
78  * Data Memory Barrier (DMB) acts as a memory barrier. It ensures that all
79  * explicit memory accesses that appear in program order before the DMB
80  * instruction are observed before any explicit memory accesses that appear
81  * in program order after the DMB instruction. It does not affect the
82  * ordering of any other instructions executing on the processor
83  *
84  * dmb st - Data memory barrier. Wait for stores to complete.
85  *
86  * Data Synchronization Barrier (DSB) acts as a special kind of memory
87  * barrier. No instruction in program order after this instruction executes
88  * until this instruction completes. This instruction completes when: (1) All
89  * explicit memory accesses before this instruction complete, and (2) all
90  * Cache, Branch predictor and TLB maintenance operations before this
91  * instruction complete.
92  *
93  * dsb sy - Data syncrhonization barrier. Assures that the CPU waits until
94  * all memory accesses are complete
95  */
96 
97 #define SP_DSB(n) __asm__ __volatile__ ("dsb sy" : : : "memory")
98 #define SP_DMB(n) __asm__ __volatile__ ("dmb st" : : : "memory")

106 /* The Type of a spinlock.
107  *
108  * ARMv6 architecture introuced the concept of exclusive accesses to memory
109  * locations in the form of the Load-Exclusive (LDREX) and Store-Exclusive
110  * (STREX) instructions in ARM and Thumb instruction sets. ARMv6K extended
111  * this to included byte, halfword, and doubleword variants of LDREX and
112  * STREX. ARMv7-M supports byte and halfwor, but not the doudleword varient
113  * (ARMv6-M does not support exlusive access)
114  *
115  * ARM architectures prior to ARMv6 supported SWP and SWPB instructions that
116  * atomically swap a 32-bit word for byte value between a register and a
117  * memory location. From the ARMv6 architecture, ARM deprecates the use
118  * of SWP and SWPB.
119  */
