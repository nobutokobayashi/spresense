62 /* Configuration ************************************************************/
63 /* CONFIG_RTC - Enables general support for a hardware RTC. Specific
64  * architectures may require other specific settings.
65  *
66  * CONFIG_RTC_DATETIME - There are two general types of RTC: (1) A simple
67  * battery backed counter that keeps the time when power is down, and (2)
68  * A full date / time RTC the provides the date and time information, often
69  * in BCD format. If CONFIG_RTC_DATETIME is selected, it specifies this
70  * second kind of RTC. In this case, the RTC is used to "seed" the normal
71  * NuttX timer and the NuttX system timer provides for higher resolution
72  * time.
73  *
74  * CONFIG_RTC_HIRES - If CONFIG_RTC_DATETIME not selected, then the simple,
75  * battery backed counter is used. There are two different implementations
76  * of such simple counters based on the time resolution of the counter:
77  * The typical RTC keeps time to resolution of 1 second, usually
78  * supporting a 32-bit time_t value. In this case, the RTC is used to
79  * "seed" the normal NuttX timer and the NuttX timer provides for higher
80  * resolution time.
81  *
82  * If CONFIG_RTC_HIRES is enabled in the NuttX configuration, then the
83  * RTC provides higher resolution time and completely replaces the system
84  * timer for purpose of date and time.
85  *
86  * CONFIG_RTC_FREQUENCY - If CONFIG_RTC_HIRES is defined, then the frequency
87  * of the high resolution RTC must be provided. If CONFIG_RTC_HIRES is
88  * not defined, CONFIG_RTC_FREQUENCY is assumed to be one.
89  *
90  * CONFIG_RTC_ALARM - Enable if the RTC hardware supports setting of an
91  * alarm. A callback function will be executed when the alarm goes off
92  *
93  * CONFIG_RTC_DRIVER - Enable building the upper-half RTC driver
94  */
95

295 /* The RTC driver is implemented as a common, upper-half character driver
296  * that provides the RTC driver structure and a lower-level, hardware
297  * specific implementation that performs the actual RTC operations.
298  *
299  * struct rtc_ops_s defines the callable interface from the common upper-
300  * half driver into lower half implementation. Each method in this
301  * structure corresponds to one of the RTC IOCTL commands. All IOCTL
302  * command logic is implemented in the lower-half driver.
303  *
304  * A NULL value should be provided for any unsupported methods.
305  */
306 
307 struct rtc_lowerhalf_s;
