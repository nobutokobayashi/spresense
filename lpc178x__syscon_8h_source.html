219 /* Flash accelerator module */
220  /* Bits 0-11: Reserved */
221 #define SYSCON_FLASHCFG_TIM_SHIFT (12) /* Bits 12-15: FLASHTIM Flash access time */
222 #define SYSCON_FLASHCFG_TIM_MASK (15 << SYSCON_FLASHCFG_TIM_SHIFT)
223 # define SYSCON_FLASHCFG_TIM_0 (0) /* 1 CPU clock <= 20 MHz CPU clock */
224 # define SYSCON_FLASHCFG_TIM_1 (1 << SYSCON_FLASHCFG_TIM_SHIFT) /* 2 CPU clock <= 40 MHz CPU clock */
225 # define SYSCON_FLASHCFG_TIM_2 (2 << SYSCON_FLASHCFG_TIM_SHIFT) /* 3 CPU clock <= 60 MHz CPU clock */
226 # define SYSCON_FLASHCFG_TIM_3 (3 << SYSCON_FLASHCFG_TIM_SHIFT) /* 4 CPU clock <= 80 MHz CPU clock */
227 # define SYSCON_FLASHCFG_TIM_4 (4 << SYSCON_FLASHCFG_TIM_SHIFT) /* 5 CPU clock <= 100 MHz CPU clock
228  * (Up to 120 Mhz for LPC1788x) */
229 # define SYSCON_FLASHCFG_TIM_5 (5 << SYSCON_FLASHCFG_TIM_SHIFT) /* "safe" setting for any conditions */
230  /* Bits 16-31: Reserved */
231 /* Memory Mapping Control register (MEMMAP - 0x400F C040) */
232 
233 #define SYSCON_MEMMAP_MAP (1 << 0) /* Bit 0:
234  * 0:Boot mode. A portion of the Boot ROM is mapped to address 0.
235  * 1:User mode. The on-chip Flash memory is mapped to address 0 */
236  /* Bits 1-31: Reserved */

238

242 # define SYSCON_CLKSRCSEL_MAIN (1 << SYSCON_CLKSRCSEL_SHIFT) /* PLL0 source = main oscillator */
243  /* Bits 1-31: Reserved */

253 #define SYSCON_PLLCFG_PSEL_SHIFT (5) /* Bit 5-6: PLL Pre-Divider value */
