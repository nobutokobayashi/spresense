57 /* Clocking *************************************************************************/
58 /* Four clock sources are available on STM3220G-EVAL evaluation board for
59  * STM32F207IGH6 and RTC embedded:
60  *
61  * X1, 25 MHz crystal for ethernet PHY with socket. It can be removed when clock is
62  * provided by MCO pin of the MCU
63  * X2, 26 MHz crystal for USB OTG HS PHY
64  * X3, 32 kHz crystal for embedded RTC
65  * X4, 25 MHz crystal with socket for STM32F207IGH6 microcontroller (It can be removed
66  * from socket when internal RC clock is used.)
67  *
68  * This is the "standard" configuration as set up by arch/arm/src/stm32f40xx_rcc.c:
69  * System Clock source : PLL (HSE)
70  * SYSCLK(Hz) : 120000000 Determined by PLL configuration
71  * HCLK(Hz) : 120000000 (STM32_RCC_CFGR_HPRE)
72  * AHB Prescaler : 1 (STM32_RCC_CFGR_HPRE)
73  * APB1 Prescaler : 4 (STM32_RCC_CFGR_PPRE1)
74  * APB2 Prescaler : 2 (STM32_RCC_CFGR_PPRE2)
75  * HSE Frequency(Hz) : 25000000 (STM32_BOARD_XTAL)
76  * PLLM : 25 (STM32_PLLCFG_PLLM)
77  * PLLN : 240 (STM32_PLLCFG_PLLN)
78  * PLLP : 2 (STM32_PLLCFG_PLLP)
79  * PLLQ : 5 (STM32_PLLCFG_PLLQ)
80  * Main regulator output voltage : Scale1 mode Needed for high speed SYSCLK
81  * Flash Latency(WS) : 5
82  * Prefetch Buffer : OFF
83  * Instruction cache : ON
84  * Data cache : ON
85  * Require 48MHz for USB OTG FS, : Enabled
86  * SDIO and RNG clock
87  */
88

102 /* Main PLL Configuration.
103  *
104  * PLL source is HSE
105  * PLL_VCO = (STM32_HSE_FREQUENCY / PLLM) * PLLN
106  * = (25,000,000 / 25) * 240
107  * = 240,000,000
108  * SYSCLK = PLL_VCO / PLLP
