57 /* Clocking *************************************************************************/
58 /* Four different clock sources can be used to drive the system clock (SYSCLK):
59  *
60  * - HSI high-speed internal oscillator clock
61  * Generated from an internal 16 MHz RC oscillator
62  * - HSE high-speed external oscillator clock
63  * Normally driven by an external crystal (X3). However, this crystal is not fitted
64  * on the STM32L-Discovery board.
65  * - PLL clock
66  * - MSI multispeed internal oscillator clock
67  * The MSI clock signal is generated from an internal RC oscillator. Seven frequency
68  * ranges are available: 65.536 kHz, 131.072 kHz, 262.144 kHz, 524.288 kHz, 1.048 MHz,
69  * 2.097 MHz (default value) and 4.194 MHz.
70  *
71  * The devices have the following two secondary clock sources
72  * - LSI low-speed internal RC clock
73  * Drives the watchdog and RTC. Approximately 37KHz
74  * - LSE low-speed external oscillator clock
75  * Driven by 32.768KHz crystal (X2) on the OSC32_IN and OSC32_OUT pins.
76  */
77

81 #define STM32_HSE_FREQUENCY STM32_BOARD_XTAL
82 #define STM32_MSI_FREQUENCY 2097000 /* Default is approximately 2.097Mhz */
83 #define STM32_LSI_FREQUENCY 37000 /* Approximately 37KHz */

86 /* PLL Configuration
87  *
88  * - PLL source is HSI -> 16MHz input (nominal)
89  * - PLL multipler is 6 -> 96MHz PLL VCO clock output (for USB)
90  * - PLL output divider 3 -> 32MHz divided down PLL VCO clock output
91  *
92  * Resulting SYSCLK frequency is 16MHz x 6 / 3 = 32MHz
93  *
94  * USB/SDIO:
95  * If the USB or SDIO interface is used in the application, the PLL VCO
96  * clock (defined by STM32_CFGR_PLLMUL) must be programmed to output a 96
97  * MHz frequency. This is required to provide a 48 MHz clock to the USB or
98  * SDIO (SDIOCLK or USBCLK = PLLVCO/2).
99  * SYSCLK
100  * The system clock is derived from the PLL VCO divided by the output division factor.
101  * Limitations:
102  * 96 MHz as PLLVCO when the product is in range 1 (1.8V),
103  * 48 MHz as PLLVCO when the product is in range 2 (1.5V),
104  * 24 MHz when the product is in range 3 (1.2V).
105  * Output division to avoid exceeding 32 MHz as SYSCLK.
106  * The minimum input clock frequency for PLL is 2 MHz (when using HSE as PLL source).
107  */
108

164 /* LED definitions ******************************************************************/
165 /* The STM32L-Discovery board has four LEDs. Two of these are controlled by
166  * logic on the board and are not available for software control:
167  *
168  * LD1 COM: LD2 default status is red. LD2 turns to green to indicate that
169  * communications are in progress between the PC and the ST-LINK/V2.
170  * LD2 PWR: Red LED indicates that the board is powered.
171  *
172  * And two LEDs can be controlled by software:
173  *
174  * User LD3: Green LED is a user LED connected to the I/O PB7 of the STM32L152
175  * MCU.
176  * User LD4: Blue LED is a user LED connected to the I/O PB6 of the STM32L152
177  * MCU.
178  *
179  * If CONFIG_ARCH_LEDS is not defined, then the user can control the LEDs in any
180  * way. The following definitions are used to access individual LEDs.
181  */
182
