113 /* Representation of a local connection. There are four types of
114  * connection structures:
115  *
116  * 1. Server. A SOCK_STREAM that only listens for and accepts
117  * connections from server.
118  * 2. Client. A SOCK_STREAM peer that connects via the server.
119  * 3. Peer. A connected SOCK_STREAM that sends() and recvs() packets.
120  * May either be the client that connect with the server of the
121  * new peer connect generated with the connection was accepted by
122  * the server.
123  *
124  * And
125  *
126  * 4. Connectionless. Like a peer but using a connectionless datagram
127  * style of communication. SOCK_DRAM support has not yet been
128  * implemented.
129  */
130 
131 struct local_conn_s
132 {
133  /* lc_node supports a doubly linked list: Listening SOCK_STREAM servers
134  * will be linked into a list of listeners; SOCK_STREAM clients will be
135  * linked to the lc_waiters and lc_conn lists.
136  */
137 
138  dq_entry_t lc_node; /* Supports a doubly linked list */
139 
140  /* Fields common to SOCK_STREAM and SOCK_DGRAM */
141 
142  uint8_t lc_crefs; /* Reference counts on this instance */
143  uint8_t lc_proto; /* SOCK_STREAM or SOCK_DGRAM */
144  uint8_t lc_type; /* See enum local_type_e */
145  uint8_t lc_state; /* See enum local_state_e */
146  int16_t lc_infd; /* File descriptor of read-only FIFO (peers) */
147  int16_t lc_outfd; /* File descriptor of write-only FIFO (peers) */
148  char lc_path[UNIX_PATH_MAX]; /* Path assigned by bind() */
149  int32_t lc_instance_id; /* Connection instance ID for stream
150  * server<->client connection pair */
151 
152 #ifdef CONFIG_NET_LOCAL_STREAM
153  /* SOCK_STREAM fields common to both client and server */
154 
155  sem_t lc_waitsem; /* Use to wait for a connection to be accepted */
156 
157 #ifdef HAVE_LOCAL_POLL
158  /* The following is a list if poll structures of threads waiting for
159  * socket accept events.
160  */
161 
162  struct pollfd *lc_accept_fds[LOCAL_ACCEPT_NPOLLWAITERS];
163 #endif

165  /* Union of fields unique to SOCK_STREAM client, server, and connected
166  * peers.
167  */
168 
169  union
170  {
171  /* Fields unique to the SOCK_STREAM server side */
172 
173  struct
174  {
175  uint8_t lc_pending; /* Number of pending connections */
176  uint8_t lc_backlog; /* Maximum number of pending connections */
177  dq_queue_t lc_waiters; /* List of connections waiting to be accepted */
178  } server;
179 
180  /* Fields unique to the connecting client side */
181 
182  struct
183  {
184  uint16_t lc_remaining; /* (For binary compatibility with peer) */
185  volatile int lc_result; /* Result of the connection operation (client)*/
186  } client;
187 
188  /* Fields common to connected peers (connected or accepted) */
189 
190  struct
191  {
192  uint16_t lc_remaining; /* Bytes remaining in the incoming stream */
193  } peer;
194  } u;
195 #endif /* CONFIG_NET_LOCAL_STREAM */
196 };
